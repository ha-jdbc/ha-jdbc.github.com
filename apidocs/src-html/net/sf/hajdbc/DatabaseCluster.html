018package net.sf.hajdbc;
019
020import java.util.concurrent.ExecutorService;
021import java.util.concurrent.ThreadFactory;
022
023import net.sf.hajdbc.balancer.Balancer;
024import net.sf.hajdbc.cache.DatabaseMetaDataCache;
025import net.sf.hajdbc.codec.Decoder;
026import net.sf.hajdbc.dialect.Dialect;
027import net.sf.hajdbc.durability.Durability;
028import net.sf.hajdbc.lock.LockManager;
029import net.sf.hajdbc.state.StateManager;
030import net.sf.hajdbc.tx.TransactionIdentifierFactory;
031
032/**
033 * @author Paul Ferraro
034 * @param <Z> either java.sql.Driver or javax.sql.DataSource
035 * @param <D> database implementation
036 */
037public interface DatabaseCluster<Z, D extends Database<Z>> extends Lifecycle
038{
039        /**
040         * Returns the identifier of this cluster.
041         * @return an identifier
042         */
043        String getId();
044        
045        /**
046         * Activates the specified database
047         * @param database a database descriptor
048         * @param manager a state manager
049         * @return true, if the database was activated, false it was already active
050         */
051        boolean activate(D database, StateManager manager);
052        
053        /**
054         * Deactivates the specified database
055         * @param database a database descriptor
056         * @param manager a state manager
057         * @return true, if the database was deactivated, false it was already inactive
058         */
059        boolean deactivate(D database, StateManager manager);
060        
061        /**
062         * Returns the database identified by the specified id
063         * @param id a database identifier
064         * @return a database descriptor
065         * @throws IllegalArgumentException if no database exists with the specified identifier
066         */
067        D getDatabase(String id);
068        
069        /**
070         * Returns the Balancer implementation used by this database cluster.
071         * @return an implementation of <code>Balancer</code>
072         */
073        Balancer<Z, D> getBalancer();
074        
075        TransactionMode getTransactionMode();
076        
077        ExecutorService getExecutor();
078        
079        /**
080         * Returns an executor service used to execute end transaction statements.
081         * @return an implementation of <code>ExecutorService</code>
082         */
083//      ExecutorService getEndTransactionExecutor();
084        
085        /**
086         * Returns an executor service used to execute transactional database writes.
087         * @return an implementation of <code>ExecutorService</code>
088         */
089//      ExecutorService getTransactionalExecutor();
090        
091        /**
092         * Returns an executor service used to execute non-transactional database writes.
093         * @return an implementation of <code>ExecutorService</code>
094         */
095//      ExecutorService getNonTransactionalExecutor();
096        
097        /**
098         * Returns a dialect capable of returning database vendor specific values.
099         * @return an implementation of <code>Dialect</code>
100         */
101        Dialect getDialect();
102        
103        /**
104         * Returns a LockManager capable of acquiring named read/write locks on the specific objects in this database cluster.
105         * @return a LockManager implementation
106         */
107        LockManager getLockManager();
108        
109        /**
110         * Returns a StateManager for persisting database cluster state.
111         * @return a StateManager implementation
112         */
113        StateManager getStateManager();
114        
115        /**
116         * Returns a DatabaseMetaData cache.
117         * @return a <code>DatabaseMetaDataCache</code> implementation
118         */
119        DatabaseMetaDataCache<Z, D> getDatabaseMetaDataCache();
120        
121        /**
122         * Indicates whether or not sequence detection is enabled for this cluster.
123         * @return true, if sequence detection is enabled, false otherwise.
124         */
125        boolean isSequenceDetectionEnabled();
126        
127        /**
128         * Indicates whether or not identity column detection is enabled for this cluster.
129         * @return true, if identity column detection is enabled, false otherwise.
130         */
131        boolean isIdentityColumnDetectionEnabled();
132        
133        /**
134         * Indicates whether or not non-deterministic CURRENT_DATE SQL functions will be evaluated to deterministic static values.
135         * @return true, if temporal SQL replacement is enabled, false otherwise.
136         */
137        boolean isCurrentDateEvaluationEnabled();
138        
139        /**
140         * Indicates whether or not non-deterministic CURRENT_TIME functions will be evaluated to deterministic static values.
141         * @return true, if temporal SQL replacement is enabled, false otherwise.
142         */
143        boolean isCurrentTimeEvaluationEnabled();
144        
145        /**
146         * Indicates whether or not non-deterministic CURRENT_TIMESTAMP functions will be evaluated to deterministic static values.
147         * @return true, if temporal SQL replacement is enabled, false otherwise.
148         */
149        boolean isCurrentTimestampEvaluationEnabled();
150        
151        /**
152         * Indicates whether or not non-deterministic RAND() functions will be replaced by evaluated to static values.
153         * @return true, if temporal SQL replacement is enabled, false otherwise.
154         */
155        boolean isRandEvaluationEnabled();
156        
157        /**
158         * Indicates whether or not this cluster is active, i.e. started, but not yet stopped.
159         * @return true, if this cluster is active, false otherwise.
160         */
161        boolean isActive();
162
163        void addListener(DatabaseClusterListener listener);
164        
165        void removeListener(DatabaseClusterListener listener);
166        
167        void addSynchronizationListener(SynchronizationListener listener);
168        
169        void removeSynchronizationListener(SynchronizationListener listener);
170        
171        void addConfigurationListener(DatabaseClusterConfigurationListener<Z, D> listener);
172        
173        void removeConfigurationListener(DatabaseClusterConfigurationListener<Z, D> listener);
174        
175        Durability<Z, D> getDurability();
176        
177        ThreadFactory getThreadFactory();
178        
179        Decoder getDecoder();
180        
181        TransactionIdentifierFactory<? extends Object> getTransactionIdentifierFactory();
182}
