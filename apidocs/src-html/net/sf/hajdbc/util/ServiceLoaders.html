018package net.sf.hajdbc.util;
019
020import java.util.LinkedList;
021import java.util.List;
022import java.util.ServiceLoader;
023
024import net.sf.hajdbc.logging.Level;
025import net.sf.hajdbc.logging.Logger;
026import net.sf.hajdbc.logging.LoggerFactory;
027
028public class ServiceLoaders
029{
030        private static final Logger logger = LoggerFactory.getLogger(ServiceLoaders.class);
031        
032        public static <T> T findService(Class<T> serviceClass)
033        {
034                for (T service: ServiceLoader.load(serviceClass, serviceClass.getClassLoader()))
035                {
036                        return service;
037                }
038                return null;
039        }
040
041        public static <T> T findRequiredService(Class<T> serviceClass)
042        {
043                T service = findService(serviceClass);
044                if (service == null)
045                {
046                        throw new IllegalStateException(String.format("No %s found", serviceClass.getName()));
047                }
048                return service;
049        }
050
051        public static <T> T findService(Matcher<T> matcher, Class<T> serviceClass)
052        {
053                List<T> matches = new LinkedList<T>();
054
055                for (T service: ServiceLoader.load(serviceClass, serviceClass.getClassLoader()))
056                {
057                        if (matcher.matches(service))
058                        {
059                                matches.add(service);
060                        }
061                }
062                
063                if (matches.size() > 1)
064                {
065                        logger.log(Level.WARN, "Multiple {0} found matching {1}: {2}", serviceClass.getName(), matcher, matches);
066                }
067                
068                return !matches.isEmpty() ? matches.get(0) : null;
069        }
070
071        public static <T> T findRequiredService(Matcher<T> matcher, Class<T> serviceClass)
072        {
073                T service = findService(matcher, serviceClass);
074                if (service == null)
075                {
076                        throw new IllegalArgumentException(String.format("No %s found matching %s", serviceClass.getName(), matcher));
077                }
078                return service;
079        }
080
081        private ServiceLoaders()
082        {
083                // Hide
084        }
085}
