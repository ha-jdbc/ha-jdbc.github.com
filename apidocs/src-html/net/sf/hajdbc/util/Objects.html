018package net.sf.hajdbc.util;
019
020import java.io.ByteArrayInputStream;
021import java.io.ByteArrayOutputStream;
022import java.io.IOException;
023import java.io.ObjectInput;
024import java.io.ObjectInputStream;
025import java.io.ObjectOutput;
026import java.io.ObjectOutputStream;
027import java.security.AccessController;
028import java.security.PrivilegedAction;
029import java.util.Arrays;
030
031/**
032 * Object utility methods.
033 * @author Paul Ferraro
034 */
035public class Objects
036{
037        /**
038         * Compares two objects for equality.
039         * If the objects are arrays, then an array comparison is performed.
040         * @param object1 an object
041         * @param object2 another object
042         * @return true, if the objects are equal, false otherwise
043         */
044        public static boolean equals(Object object1, Object object2)
045        {
046                if ((object1 == null) || (object2 == null)) return object1 == object2;
047
048                if (object1.getClass().isArray() && object2.getClass().isArray())
049                {
050                        if ((object1 instanceof boolean[]) && (object2 instanceof boolean[]))
051                        {
052                                return Arrays.equals((boolean[]) object1, (boolean[]) object2);
053                        }
054                        if ((object1 instanceof byte[]) && (object2 instanceof byte[]))
055                        {
056                                return Arrays.equals((byte[]) object1, (byte[]) object2);
057                        }
058                        if ((object1 instanceof char[]) && (object2 instanceof char[]))
059                        {
060                                return Arrays.equals((char[]) object1, (char[]) object2);
061                        }
062                        if ((object1 instanceof double[]) && (object2 instanceof double[]))
063                        {
064                                return Arrays.equals((double[]) object1, (double[]) object2);
065                        }
066                        if ((object1 instanceof float[]) && (object2 instanceof float[]))
067                        {
068                                return Arrays.equals((float[]) object1, (float[]) object2);
069                        }
070                        if ((object1 instanceof int[]) && (object2 instanceof int[]))
071                        {
072                                return Arrays.equals((int[]) object1, (int[]) object2);
073                        }
074                        if ((object1 instanceof long[]) && (object2 instanceof long[]))
075                        {
076                                return Arrays.equals((long[]) object1, (long[]) object2);
077                        }
078                        if ((object1 instanceof short[]) && (object2 instanceof short[]))
079                        {
080                                return Arrays.equals((short[]) object1, (short[]) object2);
081                        }
082                        return Arrays.equals((Object[]) object1, (Object[]) object2);
083                }
084
085                return object1.equals(object2);
086        }
087        
088        /**
089         * Serializes the specified object.
090         * @param object the serialization target
091         * @return a serialized object
092         */
093        public static byte[] serialize(Object object)
094        {
095                if (object == null) return null;
096                
097                ByteArrayOutputStream bytes = new ByteArrayOutputStream();
098                
099                try
100                {
101                        ObjectOutput output = new ObjectOutputStream(bytes);
102                        try
103                        {
104                                output.writeObject(object);
105                                output.flush();
106                                
107                                return bytes.toByteArray();
108                        }
109                        finally
110                        {
111                                Resources.close(output);
112                        }
113                }
114                catch (IOException e)
115                {
116                        throw new IllegalStateException(e);
117                }
118        }
119        
120        /**
121         * Deserializes the specified bytes into the object of the specified type using HA-JDBC's classloader.
122         * @param bytes a serialized object
123         * @return a deserialized object
124         */
125        public static <T> T deserialize(byte[] bytes)
126        {
127                return deserialize(bytes, Objects.class.getClassLoader());
128        }
129        
130        /**
131         * Deserializes the specified bytes into the object of the specified type.
132         * @param bytes a serialized object
133         * @param loader the classloader used during deserialization
134         * @return a deserialized object
135         */
136        public static <T> T deserialize(byte[] bytes, ClassLoader loader)
137        {
138                if (bytes == null) return null;
139                
140                try
141                {
142                        ObjectInput input = new ObjectInputStream(new ByteArrayInputStream(bytes));
143                        try
144                        {
145                                return readObject(input, loader);
146                        }
147                        finally
148                        {
149                                Resources.close(input);
150                        }
151                }
152                catch (IOException e)
153                {
154                        throw new IllegalStateException(e);
155                }
156        }
157        
158        /**
159         * Reads an object from the input stream using HA-JDBC's classloader.
160         * @param input an input stream
161         * @return a deserialized object
162         */
163        public static <T> T readObject(ObjectInput input)
164        {
165                return readObject(input, Objects.class.getClassLoader());
166        }
167        
168        /**
169         * Reads an object from the input stream using the specified classloader.
170         * @param input an input stream
171         * @param loader a classloader
172         * @return a deserialized object
173         */
174        @SuppressWarnings("unchecked")
175        public static <T> T readObject(ObjectInput input, ClassLoader loader)
176        {
177                ClassLoader originalLoader = getThreadContextClassLoader();
178                
179                setThreadContextClassLoader(loader);
180                
181                try
182                {
183                        return (T) input.readObject();
184                }
185                catch (ClassNotFoundException e)
186                {
187                        throw new IllegalStateException(e);
188                }
189                catch (IOException e)
190                {
191                        throw new IllegalStateException(e);
192                }
193                finally
194                {
195                        setThreadContextClassLoader(originalLoader);
196                }
197        }
198        
199        private Objects()
200        {
201                // Hide
202        }
203        
204        private static ClassLoader getThreadContextClassLoader()
205        {
206                PrivilegedAction<ClassLoader> action = new PrivilegedAction<ClassLoader>()
207                {
208                        @Override
209                        public ClassLoader run()
210                        {
211                                return Thread.currentThread().getContextClassLoader();
212                        }
213                };
214                return AccessController.doPrivileged(action);
215        }
216        
217        private static void setThreadContextClassLoader(final ClassLoader loader)
218        {
219                PrivilegedAction<Void> action = new PrivilegedAction<Void>()
220                {
221                        @Override
222                        public Void run()
223                        {
224                                Thread.currentThread().setContextClassLoader(loader);
225                                return null;
226                        }
227                };
228                AccessController.doPrivileged(action);
229        }
230}
