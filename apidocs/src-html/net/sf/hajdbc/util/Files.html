001package net.sf.hajdbc.util;
002
003import java.io.File;
004import java.io.IOException;
005import java.security.AccessController;
006import java.security.PrivilegedAction;
007
008public class Files
009{
010        public static File createTempFile(final String suffix) throws IOException
011        {
012                PrivilegedAction<File> action = new PrivilegedAction<File>()
013                {
014                        @Override
015                        public File run()
016                        {
017                                try
018                                {
019                                        return File.createTempFile("ha-jdbc_", suffix);
020                                }
021                                catch (IOException e)
022                                {
023                                        throw new PrivilegedIOException(e);
024                                }
025                        }
026                };
027                try
028                {
029                        return AccessController.doPrivileged(action);
030                }
031                catch (PrivilegedIOException e)
032                {
033                        throw e.getException();
034                }
035        }
036        
037        public static void delete(final File file)
038        {
039                PrivilegedAction<Void> action = new PrivilegedAction<Void>()
040                {
041                        @Override
042                        public Void run()
043                        {
044                                if (!file.delete())
045                                {
046                                        file.deleteOnExit();
047                                }
048                                return null;
049                        }
050                };
051                AccessController.doPrivileged(action);
052        }
053        private static class PrivilegedIOException extends RuntimeException
054        {
055                private static final long serialVersionUID = 7017527313040459676L;
056                private final IOException e;
057                
058                PrivilegedIOException(IOException e)
059                {
060                        this.e = e;
061                }
062                
063                IOException getException()
064                {
065                        return this.e;
066                }
067        }
068}
