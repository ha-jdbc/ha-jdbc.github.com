001package net.sf.hajdbc.util;
002
003import java.io.Closeable;
004import java.io.IOException;
005import java.io.ObjectInput;
006import java.io.ObjectOutput;
007import java.sql.Connection;
008import java.sql.ResultSet;
009import java.sql.SQLException;
010import java.sql.Statement;
011
012import net.sf.hajdbc.logging.Level;
013import net.sf.hajdbc.logging.Logger;
014import net.sf.hajdbc.logging.LoggerFactory;
015
016public class Resources
017{
018        private static final Logger logger = LoggerFactory.getLogger(Resources.class);
019        
020        public static void close(Closeable resource)
021        {
022                try
023                {
024                        resource.close();
025                }
026                catch (IOException e)
027                {
028                        logger.log(Level.WARN, e);
029                }
030        }
031        
032        public static void close(ObjectOutput output)
033        {
034                try
035                {
036                        output.close();
037                }
038                catch (IOException e)
039                {
040                        logger.log(Level.WARN, e);
041                }
042        }
043        
044        public static void close(ObjectInput input)
045        {
046                try
047                {
048                        input.close();
049                }
050                catch (IOException e)
051                {
052                        logger.log(Level.WARN, e);
053                }
054        }
055        
056        public static void close(Connection connection)
057        {
058                try
059                {
060                        connection.close();
061                }
062                catch (SQLException e)
063                {
064                        logger.log(Level.WARN, e);
065                }
066        }
067        
068        public static void close(Statement statement)
069        {
070                try
071                {
072                        statement.close();
073                }
074                catch (SQLException e)
075                {
076                        logger.log(Level.WARN, e);
077                }
078        }
079        
080        public static void close(ResultSet resultSet)
081        {
082                try
083                {
084                        resultSet.close();
085                }
086                catch (SQLException e)
087                {
088                        logger.log(Level.WARN, e);
089                }
090        }
091        
092        private Resources()
093        {
094                // Hide
095        }
096}
