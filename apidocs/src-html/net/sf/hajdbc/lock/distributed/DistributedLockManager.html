045/**
046 * @author Paul Ferraro
047 */
048public class DistributedLockManager implements LockManager, LockCommandContext, Stateful, MembershipListener
049{
050        final CommandDispatcher<LockCommandContext> dispatcher;
051        
052        private final LockManager lockManager;
053        private final ConcurrentMap<Member, Map<LockDescriptor, Lock>> remoteLockDescriptorMap = new ConcurrentHashMap<Member, Map<LockDescriptor, Lock>>();
054        
055        public <Z, D extends Database<Z>> DistributedLockManager(DatabaseCluster<Z, D> cluster, CommandDispatcherFactory dispatcherFactory) throws Exception
056        {
057                this.lockManager = cluster.getLockManager();
058                LockCommandContext context = this;
059                this.dispatcher = dispatcherFactory.createCommandDispatcher(cluster.getId() + ".lock", context, this, this);
060        }
061        
062        /**
063         * {@inheritDoc}
064         * @see net.sf.hajdbc.lock.LockManager#readLock(java.lang.String)
065         */
066        @Override
067        public Lock readLock(String id)
068        {
069                return this.lockManager.readLock(id);
070        }
071
072        /**
073         * {@inheritDoc}
074         * @see net.sf.hajdbc.lock.LockManager#writeLock(java.lang.String)
075         */
076        @Override
077        public Lock writeLock(String id)
078        {
079                return this.getDistibutedLock(new RemoteLockDescriptorImpl(id, LockType.WRITE, this.dispatcher.getLocal()));
080        }
081
082        /**
083         * {@inheritDoc}
084         * @see net.sf.hajdbc.lock.distributed.LockCommandContext#getLock(net.sf.hajdbc.lock.distributed.LockDescriptor)
085         */
086        @Override
087        public Lock getLock(LockDescriptor lock)
088        {
089                String id = lock.getId();
090                
091                switch (lock.getType())
092                {
093                        case READ:
094                        {
095                                return this.lockManager.readLock(id);
096                        }
097                        case WRITE:
098                        {
099                                return this.lockManager.writeLock(id);
100                        }
101                        default:
102                        {
103                                throw new IllegalStateException();
104                        }
105                }
106        }
107
108        /**
109         * {@inheritDoc}
110         * @see net.sf.hajdbc.lock.distributed.LockCommandContext#getDistibutedLock(net.sf.hajdbc.lock.distributed.RemoteLockDescriptor)
111         */
112        @Override
113        public Lock getDistibutedLock(RemoteLockDescriptor descriptor)
114        {
115                return new DistributedLock(descriptor, this.getLock(descriptor), this.dispatcher);
116        }
117
118        /**
119         * {@inheritDoc}
120         * @see net.sf.hajdbc.Lifecycle#start()
121         */
122        @Override
123        public void start() throws Exception
124        {
125                this.lockManager.start();
126                this.dispatcher.start();
127        }
128
129        /**
130         * {@inheritDoc}
131         * @see net.sf.hajdbc.Lifecycle#stop()
132         */
133        @Override
134        public void stop()
135        {
136                this.dispatcher.stop();
137                this.lockManager.stop();
138        }
139
140        /**
141         * {@inheritDoc}
142         * @see net.sf.hajdbc.lock.distributed.LockCommandContext#getRemoteLocks(net.sf.hajdbc.distributed.Remote)
143         */
144        @Override
145        public Map<LockDescriptor, Lock> getRemoteLocks(Remote remote)
146        {
147                return this.remoteLockDescriptorMap.get(remote.getMember());
148        }
149
150        /**
151         * {@inheritDoc}
152         * @see net.sf.hajdbc.distributed.Stateful#writeState(java.io.ObjectOutput)
153         */
154        @Override
155        public void writeState(ObjectOutput output) throws IOException
156        {
157                output.writeInt(this.remoteLockDescriptorMap.size());
158                
159                for (Map.Entry<Member, Map<LockDescriptor, Lock>> entry: this.remoteLockDescriptorMap.entrySet())
160                {
161                        output.writeObject(entry.getKey());
162                        
163                        Set<LockDescriptor> descriptors = entry.getValue().keySet();
164                        
165                        output.writeInt(descriptors.size());
166                        
167                        for (LockDescriptor descriptor: descriptors)
168                        {
169                                output.writeUTF(descriptor.getId());
170                                output.writeByte(descriptor.getType().ordinal());
171                        }
172                }
173        }
174
175        /**
176         * {@inheritDoc}
177         * @see net.sf.hajdbc.distributed.Stateful#readState(java.io.ObjectInput)
178         */
179        @Override
180        public void readState(ObjectInput input) throws IOException, ClassNotFoundException
181        {
182                // Is this valid?  or should we unlock/clear?
183                //assert this.remoteLockDescriptorMap.isEmpty();
184                
185                int size = input.readInt();
186                
187                LockType[] types = LockType.values();
188                
189                for (int i = 0; i < size; ++i)
190                {
191                        Member member = Objects.readObject(input);
192                        
193                        Map<LockDescriptor, Lock> map = new HashMap<LockDescriptor, Lock>();
194                        
195                        int locks = input.readInt();
196                        
197                        for (int j = 0; j < locks; ++j)
198                        {
199                                String id = input.readUTF();
200                                LockType type = types[input.readByte()];
201                                
202                                LockDescriptor descriptor = new RemoteLockDescriptorImpl(id, type, member);
203                                
204                                Lock lock = this.getLock(descriptor);
205                                
206                                lock.lock();
207                                
208                                map.put(descriptor, lock);
209                        }
210                        
211                        this.remoteLockDescriptorMap.put(member, map);
212                }
213        }
214
215        /**
216         * {@inheritDoc}
217         * @see net.sf.hajdbc.distributed.MembershipListener#added(net.sf.hajdbc.distributed.Member)
218         */
219        @Override
220        public void added(Member member)
221        {
222                this.remoteLockDescriptorMap.putIfAbsent(member, new HashMap<LockDescriptor, Lock>());
223        }
224
225        /**
226         * {@inheritDoc}
227         * @see net.sf.hajdbc.distributed.MembershipListener#removed(net.sf.hajdbc.distributed.Member)
228         */
229        @Override
230        public void removed(Member member)
231        {
232                Map<LockDescriptor, Lock> locks = this.remoteLockDescriptorMap.remove(member);
233                
234                if (locks != null)
235                {
236                        for (Lock lock: locks.values())
237                        {
238                                lock.unlock();
239                        }
240                }
241        }
242        
243        private static class DistributedLock implements Lock
244        {
245                private final RemoteLockDescriptor descriptor;
246                private final Lock lock;
247                private final CommandDispatcher<LockCommandContext> dispatcher;
248                
249                DistributedLock(RemoteLockDescriptor descriptor, Lock lock, CommandDispatcher<LockCommandContext> dispatcher)
250                {
251                        this.descriptor = descriptor;
252                        this.lock = lock;
253                        this.dispatcher = dispatcher;
254                }
255                
256                @Override
257                public void lock()
258                {
259                        boolean locked = false;
260                        
261                        while (!locked)
262                        {
263                                if (this.dispatcher.isCoordinator())
264                                {
265                                        this.lock.lock();
266                                        
267                                        try
268                                        {
269                                                locked = this.lockMembers();
270                                        }
271                                        finally
272                                        {
273                                                if (!locked)
274                                                {
275                                                        this.lock.unlock();
276                                                }
277                                        }
278                                }
279                                else
280                                {
281                                        locked = this.lockCoordinator(Long.MAX_VALUE);
282                                }
283                                
284                                if (!locked)
285                                {
286                                        Thread.yield();
287                                }
288                        }
289                }
290
291                @Override
292                public void lockInterruptibly() throws InterruptedException
293                {
294                        boolean locked = false;
295                        
296                        while (!locked)
297                        {
298                                if (this.dispatcher.isCoordinator())
299                                {
300                                        this.lock.lockInterruptibly();
301                                        
302                                        try
303                                        {
304                                                locked = this.lockMembers();
305                                        }
306                                        finally
307                                        {
308                                                if (!locked)
309                                                {
310                                                        this.lock.unlock();
311                                                }
312                                        }
313                                }
314                                else
315                                {
316                                        locked = this.lockCoordinator(Long.MAX_VALUE);
317                                }
318                                
319                                if (Thread.currentThread().isInterrupted())
320                                {
321                                        throw new InterruptedException();
322                                }
323                                
324                                if (!locked)
325                                {
326                                        Thread.yield();
327                                }
328                        }
329                }
330
331                @Override
332                public boolean tryLock()
333                {
334                        boolean locked = false;
335                        
336                        if (this.dispatcher.isCoordinator())
337                        {
338                                if (this.lock.tryLock())
339                                {
340                                        try
341                                        {
342                                                locked = this.lockMembers();
343                                        }
344                                        finally
345                                        {
346                                                if (!locked)
347                                                {
348                                                        this.lock.unlock();
349                                                }
350                                        }
351                                }
352                        }
353                        else
354                        {
355                                locked = this.lockCoordinator(0);
356                        }
357                        
358                        return locked;
359                }
360
361                @Override
362                public boolean tryLock(long time, TimeUnit unit) throws InterruptedException
363                {
364                        boolean locked = false;
365                        
366                        if (this.dispatcher.isCoordinator())
367                        {
368                                if (this.lock.tryLock(time, unit))
369                                {
370                                        try
371                                        {
372                                                locked = this.lockMembers();
373                                        }
374                                        finally
375                                        {
376                                                if (!locked)
377                                                {
378                                                        this.lock.unlock();
379                                                }
380                                        }
381                                }
382                        }
383                        else
384                        {
385                                locked = this.lockCoordinator(unit.toMillis(time));
386                        }
387                        
388                        return locked;
389                }
390
391                private boolean lockMembers()
392                {
393                        boolean locked = true;
394                        
395                        Map<Member, Boolean> results = this.dispatcher.executeAll(new MemberAcquireLockCommand(this.descriptor));
396                        
397                        for (Map.Entry<Member, Boolean> entry: results.entrySet())
398                        {
399                                locked &= entry.getValue();
400                        }
401                        
402                        if (!locked)
403                        {
404                                this.unlockMembers();
405                        }
406                        
407                        return locked;
408                }
409                
410                private boolean lockCoordinator(long timeout)
411                {
412                        return this.dispatcher.executeCoordinator(new CoordinatorAcquireCommand(this.descriptor, timeout));
413                }
414                
415                @Override
416                public void unlock()
417                {
418                        if (this.dispatcher.isCoordinator())
419                        {
420                                this.unlockMembers();
421                                
422                                this.lock.unlock();
423                        }
424                        else
425                        {
426                                this.unlockCoordinator();
427                        }
428                }
429                
430                private void unlockMembers()
431                {
432                        this.dispatcher.executeAll(new MemberReleaseLockCommand(this.descriptor));
433                }
434                
435                private void unlockCoordinator()
436                {
437                        this.dispatcher.executeCoordinator(new CoordinatorReleaseCommand(this.descriptor));
438                }
439
440                @Override
441                public Condition newCondition()
442                {
443                        throw new UnsupportedOperationException();
444                }
445        }
446        
447        private static class RemoteLockDescriptorImpl implements RemoteLockDescriptor
448        {
449                private static final long serialVersionUID = 1950781245453120790L;
450                
451                private final String id;
452                private transient LockType type;                
453                private final Member member;
454                
455                RemoteLockDescriptorImpl(String id, LockType type, Member member)
456                {
457                        this.id = id;
458                        this.type = type;
459                        this.member = member;
460                }
461                
462                @Override
463                public String getId()
464                {
465                        return this.id;
466                }
467
468                @Override
469                public LockType getType()
470                {
471                        return this.type;
472                }
473
474                @Override
475                public Member getMember()
476                {
477                        return this.member;
478                }
479
480                private void writeObject(ObjectOutputStream out) throws IOException
481                {
482                        out.defaultWriteObject();
483                        
484                        out.writeByte(this.type.ordinal());
485                }
486                
487                private void readObject(ObjectInputStream in) throws IOException, ClassNotFoundException
488                {
489                        in.defaultReadObject();
490                        
491                        this.type = LockType.values()[in.readByte()];
492                }
493                
494                /**
495                 * {@inheritDoc}
496                 * @see java.lang.Object#equals(java.lang.Object)
497                 */
498                @Override
499                public boolean equals(Object object)
500                {
501                        if ((object == null) || !(object instanceof RemoteLockDescriptor)) return false;
502                        
503                        String id = ((RemoteLockDescriptor) object).getId();
504                        
505                        return ((this.id != null) && (id != null)) ?  this.id.equals(id) : (this.id == id);
506                }
507
508                /**
509                 * {@inheritDoc}
510                 * @see java.lang.Object#hashCode()
511                 */
512                @Override
513                public int hashCode()
514                {
515                        return this.id != null ? this.id.hashCode() : 0;
516                }
517
518                /**
519                 * {@inheritDoc}
520                 * @see java.lang.Object#toString()
521                 */
522                @Override
523                public String toString()
524                {
525                        return String.format("%sLock(%s)", this.type.name().toLowerCase(), (this.id != null) ? this.id : "");
526                }
527        }
