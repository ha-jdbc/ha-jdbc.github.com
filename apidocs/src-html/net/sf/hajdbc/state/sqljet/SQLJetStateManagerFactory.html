001package net.sf.hajdbc.state.sqljet;
002
003import net.sf.hajdbc.Database;
004import net.sf.hajdbc.DatabaseCluster;
005import net.sf.hajdbc.state.StateManager;
006import net.sf.hajdbc.state.StateManagerFactory;
007
008public class SQLJetStateManagerFactory implements StateManagerFactory
009{
010        private static final long serialVersionUID = 8990527398117188315L;
011
012        @Override
013        public String getId()
014        {
015                return "sqlite";
016        }
017
018        @Override
019        public <Z, D extends Database<Z>> StateManager createStateManager(DatabaseCluster<Z, D> cluster)
020        {
021                return new SQLJetStateManager<Z, D>(cluster);
022        }
023}
