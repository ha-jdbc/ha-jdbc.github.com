001package net.sf.hajdbc.state.simple;
002
003import java.util.HashMap;
004import java.util.Map;
005import java.util.Set;
006import java.util.concurrent.ConcurrentHashMap;
007import java.util.concurrent.CopyOnWriteArraySet;
008
009import net.sf.hajdbc.durability.InvocationEvent;
010import net.sf.hajdbc.durability.InvokerEvent;
011import net.sf.hajdbc.state.DatabaseEvent;
012import net.sf.hajdbc.state.StateManager;
013
014public class SimpleStateManager implements StateManager
015{
016        private final Map<InvocationEvent, Map<String, InvokerEvent>> invocations = new ConcurrentHashMap<InvocationEvent, Map<String, InvokerEvent>>();
017        private final Set<String> activeDatabases = new CopyOnWriteArraySet<String>();
018
019        @Override
020        public Set<String> getActiveDatabases()
021        {
022                return this.activeDatabases;
023        }
024
025        @Override
026        public Map<InvocationEvent, Map<String, InvokerEvent>> recover()
027        {
028                return this.invocations;
029        }
030
031        @Override
032        public void setActiveDatabases(Set<String> databases)
033        {
034                this.activeDatabases.retainAll(databases);
035                this.activeDatabases.addAll(databases);
036        }
037
038        @Override
039        public void activated(DatabaseEvent event)
040        {
041                this.activeDatabases.add(event.getSource());
042        }
043
044        @Override
045        public void deactivated(DatabaseEvent event)
046        {
047                this.activeDatabases.remove(event.getSource());
048        }
049
050        @Override
051        public void afterInvocation(InvocationEvent event)
052        {
053                this.invocations.remove(event);
054        }
055
056        @Override
057        public void beforeInvocation(InvocationEvent event)
058        {
059                this.invocations.put(event, new HashMap<String, InvokerEvent>());
060        }
061
062        @Override
063        public void afterInvoker(InvokerEvent event)
064        {
065                this.invocations.get(new InvocationEventAdapter(event)).remove(event.getDatabaseId());
066        }
067
068        @Override
069        public void beforeInvoker(InvokerEvent event)
070        {
071                this.invocations.get(new InvocationEventAdapter(event)).put(event.getDatabaseId(), event);
072        }
073
074        @Override
075        public void start()
076        {
077        }
078
079        @Override
080        public void stop()
081        {
082        }
083}
