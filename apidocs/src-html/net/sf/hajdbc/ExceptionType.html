018package net.sf.hajdbc;
019
020import java.sql.SQLException;
021
022import javax.transaction.xa.XAException;
023
024import net.sf.hajdbc.util.Matcher;
025import net.sf.hajdbc.util.ServiceLoaders;
026
027/**
028 * @author Paul Ferraro
029 */
030@SuppressWarnings("rawtypes")
031public enum ExceptionType implements Matcher<ExceptionFactory>
032{
033        SQL(SQLException.class), XA(XAException.class);
034        
035        private final Class<? extends Exception> exceptionClass;
036        
037        private <E extends Exception> ExceptionType(final Class<E> exceptionClass)
038        {
039                this.exceptionClass = exceptionClass;
040        }
041        
042        @Override
043        public boolean matches(ExceptionFactory factory)
044        {
045                return this == factory.getType();
046        }
047
048        @SuppressWarnings("unchecked")
049        public <E extends Exception> ExceptionFactory<E> getExceptionFactory()
050        {
051                return ServiceLoaders.findRequiredService(this, ExceptionFactory.class);
052        }
053        
054        public static <E extends Exception> ExceptionFactory<E> getExceptionFactory(Class<E> exceptionClass)
055        {
056                for (ExceptionType type: ExceptionType.values())
057                {
058                        if (type.exceptionClass.equals(exceptionClass))
059                        {
060                                return type.getExceptionFactory();
061                        }
062                }
063                
064                throw new IllegalArgumentException(exceptionClass.getName());
065        }
066}
