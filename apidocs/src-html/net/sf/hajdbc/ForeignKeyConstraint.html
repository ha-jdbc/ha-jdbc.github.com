018package net.sf.hajdbc;
019
020import java.util.List;
021
022
023
024
025/**
026 * Represents a foreign key constraint on a table.
027 * 
028 * @author  Paul Ferraro
029 * @since   1.1
030 */
031public interface ForeignKeyConstraint extends UniqueConstraint
032{
033        /**
034         * @return the foreign table of this foreign key
035         */
036        QualifiedName getForeignTable();
037        
038        /**
039         * @return the foreign column of this foreign key
040         */
041        List<String> getForeignColumnList();
042        
043        /**
044         * @return Returns the deleteRule.
045         */
046        int getDeleteRule();
047
048        /**
049         * @return Returns the updateRule.
050         */
051        int getUpdateRule();
052
053        /**
054         * @return Returns the deferrability.
055         */
056        int getDeferrability();
057
058        /**
059         * @param deferrability The deferrability to set.
060         */
061        void setDeferrability(int deferrability);
062
063        /**
064         * @param deleteRule The deleteRule to set.
065         */
066        void setDeleteRule(int deleteRule);
067
068        /**
069         * @param foreignTable The foreignTable to set.
070         */
071        void setForeignTable(QualifiedName foreignTable);
072
073        /**
074         * @param updateRule The updateRule to set.
075         */
076        void setUpdateRule(int updateRule);
077}
