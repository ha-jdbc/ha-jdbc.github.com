018package net.sf.hajdbc;
019
020import java.io.Serializable;
021import java.util.Map;
022import java.util.concurrent.ThreadFactory;
023
024import net.sf.hajdbc.balancer.BalancerFactory;
025import net.sf.hajdbc.cache.DatabaseMetaDataCacheFactory;
026import net.sf.hajdbc.codec.DecoderFactory;
027import net.sf.hajdbc.dialect.DialectFactory;
028import net.sf.hajdbc.distributed.CommandDispatcherFactory;
029import net.sf.hajdbc.durability.DurabilityFactory;
030import net.sf.hajdbc.lock.LockManagerFactory;
031import net.sf.hajdbc.management.MBeanRegistrar;
032import net.sf.hajdbc.state.StateManagerFactory;
033import net.sf.hajdbc.tx.TransactionIdentifierFactory;
034import net.sf.hajdbc.util.concurrent.cron.CronExpression;
035
036/**
037 * @author Paul Ferraro
038 */
039public interface DatabaseClusterConfiguration<Z, D extends Database<Z>> extends Serializable
040{
041        CommandDispatcherFactory getDispatcherFactory();
042        
043        /**
044         * Returns the databases of this cluster indexed by identifier
045         * @return a map of databases
046         * @throws IllegalArgumentException if no database exists with the specified identifier
047         */
048        Map<String, D> getDatabaseMap();
049        
050        Map<String, SynchronizationStrategy> getSynchronizationStrategyMap();
051        
052        String getDefaultSynchronizationStrategy();
053        
054        /**
055         * Returns the Balancer implementation used by this database cluster.
056         * @return an implementation of <code>Balancer</code>
057         */
058        BalancerFactory getBalancerFactory();
059
060        TransactionMode getTransactionMode();
061        
062        ExecutorServiceProvider getExecutorProvider();
063        
064        /**
065         * Returns a dialect capable of returning database vendor specific values.
066         * @return an implementation of <code>Dialect</code>
067         */
068        DialectFactory getDialectFactory();
069        
070        /**
071         * Returns a StateManager for persisting database cluster state.
072         * @return a StateManager implementation
073         */
074        StateManagerFactory getStateManagerFactory();
075        
076        /**
077         * Returns a DatabaseMetaData cache.
078         * @return a <code>DatabaseMetaDataCache</code> implementation
079         */
080        DatabaseMetaDataCacheFactory getDatabaseMetaDataCacheFactory();
081
082        DurabilityFactory getDurabilityFactory();
083
084        LockManagerFactory getLockManagerFactory();
085        
086        /**
087         * Indicates whether or not sequence detection is enabled for this cluster.
088         * @return true, if sequence detection is enabled, false otherwise.
089         */
090        boolean isSequenceDetectionEnabled();
091        
092        /**
093         * Indicates whether or not identity column detection is enabled for this cluster.
094         * @return true, if identity column detection is enabled, false otherwise.
095         */
096        boolean isIdentityColumnDetectionEnabled();
097        
098        /**
099         * Indicates whether or not non-deterministic CURRENT_DATE SQL functions will be evaluated to deterministic static values.
100         * @return true, if temporal SQL replacement is enabled, false otherwise.
101         */
102        boolean isCurrentDateEvaluationEnabled();
103        
104        /**
105         * Indicates whether or not non-deterministic CURRENT_TIME functions will be evaluated to deterministic static values.
106         * @return true, if temporal SQL replacement is enabled, false otherwise.
107         */
108        boolean isCurrentTimeEvaluationEnabled();
109        
110        /**
111         * Indicates whether or not non-deterministic CURRENT_TIMESTAMP functions will be evaluated to deterministic static values.
112         * @return true, if temporal SQL replacement is enabled, false otherwise.
113         */
114        boolean isCurrentTimestampEvaluationEnabled();
115        
116        /**
117         * Indicates whether or not non-deterministic RAND() functions will be replaced by evaluated to static values.
118         * @return true, if temporal SQL replacement is enabled, false otherwise.
119         */
120        boolean isRandEvaluationEnabled();
121
122        CronExpression getFailureDetectionExpression();
123        
124        CronExpression getAutoActivationExpression();
125        
126        ThreadFactory getThreadFactory();
127        
128        DecoderFactory getDecoderFactory();
129        
130        MBeanRegistrar<Z, D> getMBeanRegistrar();
131        
132        TransactionIdentifierFactory<? extends Object> getTransactionIdentifierFactory();
133        
134        boolean isEmptyClusterAllowed();
135}
