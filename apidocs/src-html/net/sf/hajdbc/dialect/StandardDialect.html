131        protected String executeFunctionFormat()
132        {
133                StringBuilder builder = new StringBuilder("SELECT {0}");
134                
135                String dummyTable = this.dummyTable();
136                
137                if (dummyTable != null)
138                {
139                        builder.append(" FROM ").append(dummyTable);
140                }
141                
142                return builder.toString();
143        }

253        protected String executeFunction(Connection connection, String function) throws SQLException
254        {
255                Statement statement = connection.createStatement();
256                
257                try
258                {
259                        ResultSet resultSet = statement.executeQuery(this.executeFunctionSQL(function));
260                        
261                        resultSet.next();
262                        
263                        return resultSet.getString(1);
264                }
265                finally
266                {
267                        Resources.close(statement);
268                }
269        }
270
271        protected List<String> executeQuery(Connection connection, String sql) throws SQLException
272        {
273                Statement statement = connection.createStatement();
274                
275                try
276                {
277                        ResultSet resultSet = statement.executeQuery(sql);
278                        
279                        List<String> resultList = new LinkedList<String>();
280                        
281                        while (resultSet.next())
282                        {
283                                resultList.add(resultSet.getString(1));
284                        }
285                        
286                        return resultList;
287                }
288                finally
289                {
290                        Resources.close(statement);
291                }
292        }

322        /**
323         * @see net.sf.hajdbc.SequenceSupport#getSequences(java.sql.DatabaseMetaData)
324         */
325        @Override
326        public Map<QualifiedName, Integer> getSequences(DatabaseMetaData metaData) throws SQLException
327        {
328                ResultSet resultSet = metaData.getTables(Strings.EMPTY, null, Strings.ANY, new String[] { this.sequenceTableType() });
329                
330                try
331                {
332                        Map<QualifiedName, Integer> sequences = new HashMap<QualifiedName, Integer>();
333                        
334                        while (resultSet.next())
335                        {
336                                sequences.put(new QualifiedNameImpl(resultSet.getString("TABLE_SCHEM"), resultSet.getString("TABLE_NAME"), metaData.supportsSchemasInTableDefinitions(), metaData.supportsSchemasInDataManipulation()), 1);
337                        }
338                        
339                        return sequences;
340                }
341                finally
342                {
343                        Resources.close(resultSet);
344                }
345        }

473        /**
474         * {@inheritDoc}
475         * @see net.sf.hajdbc.dialect.Dialect#evaluateRand(java.lang.String)
476         */
477        @Override
478        public String evaluateRand(String sql)
479        {       
480                StringBuffer buffer = new StringBuffer();
481                Matcher matcher = this.randomPattern.matcher(sql);
482                
483                while (matcher.find())
484                {
485                        matcher.appendReplacement(buffer, Double.toString(Math.random()));
486                }
487                
488                return matcher.appendTail(buffer).toString();
489        }

492         * {@inheritDoc}
493         * @see net.sf.hajdbc.dialect.Dialect#indicatesFailure(java.sql.SQLException)
494         */
495        @Override
496        public boolean indicatesFailure(SQLException e)
497        {
498                String state = e.getSQLState();
499                Set<String> failureSQLStates = this.failureSQLStates();
500                
501                if ((state != null) && !failureSQLStates.isEmpty())
502                {
503                        return failureSQLStates.contains(state);
504                }
505                
506                return (e instanceof SQLNonTransientConnectionException);
507        }

629        protected boolean meetsRequirement(int minMajor, int minMinor)
630        {
631                Driver driver = this.findDriver();
632
633                if (driver != null)
634                {
635                        int major = driver.getMajorVersion();
636                        int minor = driver.getMinorVersion();
637                        return (major > minMajor) || ((major == minMajor) && (minor >= minMinor));
638                }
639                
640                return false;
641        }
642        
643        private Driver findDriver()
644        {
645                String url = String.format("jdbc:%s:test", this.vendorPattern());
646                
647                List<Driver> drivers = Collections.list(DriverManager.getDrivers());
648                for (Driver driver: drivers)
649                {
650                        try
651                        {
652                                if (driver.acceptsURL(url))
653                                {
654                                        return driver;
655                                }
656                        }
657                        catch (SQLException e)
658                        {
659                                // Skip
660                        }
661                }
662                return null;
663        }
