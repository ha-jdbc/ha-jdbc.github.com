248        {
249                Statement statement = connection.createStatement();
250                
251                try
252                {
253                        ResultSet resultSet = statement.executeQuery(this.executeFunctionSQL(function));
254                        
255                        resultSet.next();
256                        
257                        return resultSet.getString(1);
258                }
259                finally
260                {
261                        Resources.close(statement);
262                }
263        }

460         * {@inheritDoc}
461         * @see net.sf.hajdbc.dialect.Dialect#evaluateRand(java.lang.String)
462         */
463        @Override
464        public String evaluateRand(String sql)
465        {       
466                StringBuffer buffer = new StringBuffer();
467                Matcher matcher = this.randomPattern.matcher(sql);
468                
469                while (matcher.find())
470                {
471                        matcher.appendReplacement(buffer, Double.toString(Math.random()));
472                }
473                
474                return matcher.appendTail(buffer).toString();
475        }

483        {
484                String state = e.getSQLState();
485                Set<String> failureSQLStates = this.failureSQLStates();
486                
487                if ((state != null) && !failureSQLStates.isEmpty())
488                {
489                        return failureSQLStates.contains(state);
490                }
491                
492                return (e instanceof SQLNonTransientConnectionException);
493        }

615        protected boolean meetsRequirement(int minMajor, int minMinor)
616        {
617                Driver driver = this.findDriver();
618
619                if (driver != null)
620                {
621                        int major = driver.getMajorVersion();
622                        int minor = driver.getMinorVersion();
623                        return (major > minMajor) || ((major == minMajor) && (minor >= minMinor));
624                }
625                
626                return false;
627        }

629        private Driver findDriver()
630        {
631                String url = String.format("jdbc:%s:test", this.vendorPattern());
632                
633                List<Driver> drivers = Collections.list(DriverManager.getDrivers());
634                for (Driver driver: drivers)
635                {
636                        try
637                        {
638                                if (driver.acceptsURL(url))
639                                {
640                                        return driver;
641                                }
642                        }
643                        catch (SQLException e)
644                        {
645                                // Skip
646                        }
647                }
648                return null;
649        }

679        /**
680         * Returns the columns of the specified table.
681         * @param metaData a DatabaseMetaData implementation
682         * @param table a schema qualified table name
683         * @return a Map of column name to column properties
684         * @throws SQLException if an error occurs access DatabaseMetaData
685         */
686        @Override
687        public Map<String, ColumnProperties> getColumns(DatabaseMetaData metaData, QualifiedName table, ColumnPropertiesFactory factory) throws SQLException
688        {
689                Statement statement = metaData.getConnection().createStatement();
690                
691                try
692                {
693                        Map<String, ColumnProperties> map = new HashMap<String, ColumnProperties>();
694                        
695                        ResultSetMetaData resultSet = statement.executeQuery(String.format("SELECT * FROM %s WHERE 0=1", table.getDMLName())).getMetaData();
696                        
697                        for (int i = 1; i <= resultSet.getColumnCount(); ++i)
698                        {
699                                String column = resultSet.getColumnName(i);
700                                int type = resultSet.getColumnType(i);
701                                String nativeType = resultSet.getColumnTypeName(i);
702                                boolean autoIncrement = resultSet.isAutoIncrement(i);
703                                
704                                ColumnProperties properties = factory.createColumnProperties(column, type, nativeType, null, null, autoIncrement);
705                                map.put(properties.getName(), properties);
706                        }
707                        
708                        return map;
709                }
710                finally
711                {
712                        Resources.close(statement);
713                }
714        }

717         * Returns the primary key of the specified table.
718         * @param metaData a DatabaseMetaData implementation
719         * @param table a schema qualified table name
720         * @return a unique constraint
721         * @throws SQLException if an error occurs access DatabaseMetaData
722         */
723        @Override
724        public UniqueConstraint getPrimaryKey(DatabaseMetaData metaData, QualifiedName table, UniqueConstraintFactory factory) throws SQLException
725        {
726                ResultSet resultSet = metaData.getPrimaryKeys(this.getCatalog(metaData), table.getSchema(), table.getName());
727                
728                try
729                {
730                        UniqueConstraint constraint = null;
731
732                        while (resultSet.next())
733                        {
734                                if (constraint == null)
735                                {
736                                        constraint = factory.createUniqueConstraint(resultSet.getString("PK_NAME"), table);
737                                }
738                                
739                                constraint.getColumnList().add(resultSet.getString("COLUMN_NAME"));
740                        }
741                        
742                        return constraint;
743                }
744                finally
745                {
746                        Resources.close(resultSet);
747                }
748        }

754         * @return a Collection of foreign key constraints.
755         * @throws SQLException if an error occurs access DatabaseMetaData
756         */
757        @Override
758        public Collection<ForeignKeyConstraint> getForeignKeyConstraints(DatabaseMetaData metaData, QualifiedName table, ForeignKeyConstraintFactory factory) throws SQLException
759        {
760                ResultSet resultSet = metaData.getImportedKeys(this.getCatalog(metaData), table.getSchema(), table.getName());
761                
762                try
763                {
764                        Map<String, ForeignKeyConstraint> foreignKeyMap = new HashMap<String, ForeignKeyConstraint>();
765                        
766                        while (resultSet.next())
767                        {
768                                String name = resultSet.getString("FK_NAME");
769                                
770                                ForeignKeyConstraint foreignKey = foreignKeyMap.get(name);
771                                
772                                if (foreignKey == null)
773                                {
774                                        foreignKey = factory.createForeignKeyConstraint(name, table, factory.getQualifiedNameFactory().createQualifiedName(resultSet.getString("PKTABLE_SCHEM"), resultSet.getString("PKTABLE_NAME")), resultSet.getInt("DELETE_RULE"), resultSet.getInt("UPDATE_RULE"), resultSet.getInt("DEFERRABILITY"));
775                                        
776                                        foreignKeyMap.put(name, foreignKey);
777                                }
778                                
779                                foreignKey.getColumnList().add(resultSet.getString("FKCOLUMN_NAME"));
780                                foreignKey.getForeignColumnList().add(resultSet.getString("PKCOLUMN_NAME"));
781                        }
782                        
783                        return foreignKeyMap.values();
784                }
785                finally
786                {
787                        Resources.close(resultSet);
788                }
789        }

791        /**
792         * Returns the unique constraints on the specified table - excluding the primary key of the table.
793         * @param metaData a schema qualified table name
794         * @param table a qualified table name
795         * @param primaryKey the primary key of this table
796         * @return a Collection of unique constraints.
797         * @throws SQLException if an error occurs access DatabaseMetaData
798         */
799        @Override
800        public Collection<UniqueConstraint> getUniqueConstraints(DatabaseMetaData metaData, QualifiedName table, UniqueConstraint primaryKey, UniqueConstraintFactory factory) throws SQLException
801        {
802                ResultSet resultSet = metaData.getIndexInfo(this.getCatalog(metaData), table.getSchema(), table.getName(), true, false);
803                
804                try
805                {
806                        Map<String, UniqueConstraint> keyMap = new HashMap<String, UniqueConstraint>();
807                        
808                        while (resultSet.next())
809                        {
810                                if (resultSet.getShort("TYPE") == DatabaseMetaData.tableIndexHashed)
811                                {
812                                        String name = resultSet.getString("INDEX_NAME");
813                                        
814                                        UniqueConstraint key = keyMap.get(name);
815                                        
816                                        if (key == null)
817                                        {
818                                                key = factory.createUniqueConstraint(name, table);
819                                                
820                                                // Don't include the primary key
821                                                if (key.equals(primaryKey)) continue;
822                                                
823                                                keyMap.put(name, key);
824                                        }
825                                        
826                                        key.getColumnList().add(resultSet.getString("COLUMN_NAME"));
827                                }
828                        }
829                        return keyMap.values();
830                }
831                finally
832                {
833                        Resources.close(resultSet);
834                }
835        }

867         * {@inheritDoc}
868         * @see net.sf.hajdbc.dialect.Dialect#getTypes(java.sql.DatabaseMetaData)
869         */
870        @Override
871        public Map<Integer, Entry<String, Integer>> getTypes(DatabaseMetaData metaData) throws SQLException
872        {
873                ResultSet resultSet = metaData.getTypeInfo();
874                
875                try
876                {
877                        Map<Integer, Map.Entry<String, Integer>> types = new HashMap<Integer, Map.Entry<String, Integer>>();
878                        
879                        while (resultSet.next())
880                        {
881                                int type = resultSet.getInt("DATA_TYPE");
882                                if (!types.containsKey(type))
883                                {
884                                        String name = resultSet.getString("TYPE_NAME");
885                                        String params = resultSet.getString("CREATE_PARAMS");
886                                        types.put(type, new AbstractMap.SimpleImmutableEntry<String, Integer>(name, (params != null) ? resultSet.getInt("PRECISION") : null));
887                                }
888                        }
889                        
890                        return types;
891                }
892                finally
893                {
894                        Resources.close(resultSet);
895                }
896        }
