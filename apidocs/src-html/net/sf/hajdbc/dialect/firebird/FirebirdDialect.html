018package net.sf.hajdbc.dialect.firebird;
019
020import java.sql.DatabaseMetaData;
021import java.sql.ResultSet;
022import java.sql.SQLException;
023import java.sql.Statement;
024import java.util.HashMap;
025import java.util.Map;
026
027import net.sf.hajdbc.QualifiedName;
028import net.sf.hajdbc.SequenceSupport;
029import net.sf.hajdbc.cache.QualifiedNameImpl;
030import net.sf.hajdbc.dialect.StandardDialect;
031import net.sf.hajdbc.util.Resources;
032
033/**
034 * Dialect for <a href="firebird.sourceforge.net">Firebird</a>.
035 * @author Paul Ferraro
036 */
037@SuppressWarnings("nls")
038public class FirebirdDialect extends StandardDialect
039{
040        /**
041         * {@inheritDoc}
042         * @see net.sf.hajdbc.dialect.StandardDialect#vendorPattern()
043         */
044        @Override
045        protected String vendorPattern()
046        {
047                return "firebird";
048        }
049
050        /**
051         * @see net.sf.hajdbc.dialect.StandardDialect#dummyTable()
052         */
053        @Override
054        protected String dummyTable()
055        {
056                return "RDB$DATABASE";
057        }
058
059        /**
060         * Firebird 2.0 will support standard syntax.  Until then...
061         * @see net.sf.hajdbc.dialect.StandardDialect#alterSequenceFormat()
062         */
063        @Override
064        protected String alterSequenceFormat()
065        {
066                return "SET GENERATOR {0} TO {1}";
067        }
068
069        /**
070         * {@inheritDoc}
071         * @see net.sf.hajdbc.dialect.StandardDialect#getSequenceSupport()
072         */
073        @Override
074        public SequenceSupport getSequenceSupport()
075        {
076                return this;
077        }
078
079        /**
080         * @see net.sf.hajdbc.dialect.StandardDialect#getSequences(java.sql.DatabaseMetaData)
081         */
082        @Override
083        public Map<QualifiedName, Integer> getSequences(DatabaseMetaData metaData) throws SQLException
084        {
085                Statement statement = metaData.getConnection().createStatement();
086                
087                try
088                {
089                        ResultSet resultSet = statement.executeQuery("SELECT RDB$GENERATOR_NAME FROM RDB$GENERATORS");
090                        
091                        Map<QualifiedName, Integer> sequences = new HashMap<QualifiedName, Integer>();
092                        
093                        while (resultSet.next())
094                        {
095                                sequences.put(new QualifiedNameImpl(resultSet.getString(1)), 1);
096                        }
097                        
098                        return sequences;
099                }
100                finally
101                {
102                        Resources.close(statement);
103                }
104        }
105
106        /**
107         * Firebird 2.0 will support standard syntax.  Until then...
108         * @see net.sf.hajdbc.dialect.StandardDialect#sequencePattern()
109         */
110        @Override
111        protected String sequencePattern()
112        {
113                return "GEN_ID\\s*\\(\\s*([^\\s,]+)\\s*,\\s*\\d+\\s*\\)";
114        }
115
116        /**
117         * @see net.sf.hajdbc.dialect.StandardDialect#selectForUpdatePattern()
118         */
119        @Override
120        protected String selectForUpdatePattern()
121        {
122                return "SELECT\\s+.+\\s+WITH\\s+LOCK";
123        }
124
125        /**
126         * @see net.sf.hajdbc.dialect.StandardDialect#nextSequenceValueFormat()
127         */
128        @Override
129        protected String nextSequenceValueFormat()
130        {
131                return "GEN_ID({0}, 1)";
132        }
133}
