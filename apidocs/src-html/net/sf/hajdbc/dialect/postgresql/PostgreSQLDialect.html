023import java.sql.ResultSet;
024import java.sql.SQLException;
025import java.sql.Statement;
026import java.sql.Types;
027import java.util.ArrayList;
028import java.util.List;
029import java.util.regex.Pattern;
030
031import net.sf.hajdbc.ColumnProperties;
032import net.sf.hajdbc.ConnectionProperties;
033import net.sf.hajdbc.DumpRestoreSupport;

041/**
042 * Dialect for <a href="http://postgresql.org">PostgreSQL</a>.
043 * @author  Paul Ferraro
044 * @since   1.1
045 */
046@SuppressWarnings("nls")
047public class PostgreSQLDialect extends StandardDialect implements DumpRestoreSupport
048{
049        private static final File PASSWORD_FILE = new File(String.format("%s%s.pgpass", Strings.USER_HOME, Strings.FILE_SEPARATOR));
050        
051        /**
052         * {@inheritDoc}
053         * @see net.sf.hajdbc.dialect.StandardDialect#vendorPattern()
054         */
055        @Override
056        protected String vendorPattern()
057        {
058                return "postgresql";
059        }
060
061        /**
062         * PostgreSQL uses a schema search path to locate unqualified table names.
063         * The default search path is [$user,public], where $user is the current user.
064         * @see net.sf.hajdbc.dialect.StandardDialect#getDefaultSchemas(java.sql.DatabaseMetaData)
065         */
066        @Override
067        public List<String> getDefaultSchemas(DatabaseMetaData metaData) throws SQLException
068        {
069                Connection connection = metaData.getConnection();
070                Statement statement = connection.createStatement();
071                
072                try
073                {
074                        ResultSet resultSet = statement.executeQuery("SHOW search_path");
075                        
076                        resultSet.next();
077                        
078                        String[] schemas = resultSet.getString(1).split(Strings.COMMA);
079                        
080                        List<String> schemaList = new ArrayList<String>(schemas.length);
081                        
082                        for (String schema: schemas)
083                        {
084                                schemaList.add(schema.equals("$user") ? metaData.getUserName() : schema);
085                        }
086                        
087                        return schemaList;
088                }
089                finally
090                {
091                        Resources.close(statement);
092                }
093        }
094
095        /**
096         * PostgreSQL uses the native type OID to identify BLOBs.
097         * However the JDBC driver incomprehensibly maps OIDs to INTEGERs.
098         * The PostgreSQL JDBC folks claim this intentional.
099         * @see net.sf.hajdbc.dialect.StandardDialect#getColumnType(net.sf.hajdbc.ColumnProperties)
100         */
101        @Override
102        public int getColumnType(ColumnProperties properties)
103        {
104                return properties.getNativeType().equalsIgnoreCase("oid") ? Types.BLOB : properties.getType();
105        }
106
107        /**
108         * Versions &gt;=8.1 of the PostgreSQL JDBC driver return incorrect values for DatabaseMetaData.getExtraNameCharacters().
109         * @see net.sf.hajdbc.dialect.StandardDialect#getIdentifierPattern(java.sql.DatabaseMetaData)
110         */
111        @Override
112        public Pattern getIdentifierPattern(DatabaseMetaData metaData) throws SQLException
113        {
114                if ((metaData.getDriverMajorVersion() >= 8) && (metaData.getDriverMinorVersion() >= 1))
115                {
116                        return Pattern.compile("[A-Za-z\\0200-\\0377_][A-Za-z\\0200-\\0377_0-9\\$]*");
117                }
118                
119                return super.getIdentifierPattern(metaData);
120        }
121
122        /**
123         * {@inheritDoc}
124         * @see net.sf.hajdbc.dialect.StandardDialect#getSequenceSupport()
125         */
126        @Override
127        public SequenceSupport getSequenceSupport()
128        {
129                return this;
130        }
131
132        /**
133         * {@inheritDoc}
134         * @see net.sf.hajdbc.dialect.StandardDialect#getIdentityColumnSupport()
135         */
136        @Override
137        public IdentityColumnSupport getIdentityColumnSupport()
138        {
139                return this;
140        }
141
142        /**
143         * @see net.sf.hajdbc.dialect.StandardDialect#truncateTableFormat()
144         */
145        @Override
146        protected String truncateTableFormat()
147        {
148                return "TRUNCATE TABLE {0}";
149        }
150
151        /**
152         * @see net.sf.hajdbc.dialect.StandardDialect#sequencePattern()
153         */
154        @Override
155        protected String sequencePattern()
156        {
157                return "(?:CURR|NEXT)VAL\\s*\\(\\s*'([^']+)'\\s*\\)";
158        }
159
160        /**
161         * @see net.sf.hajdbc.dialect.StandardDialect#nextSequenceValueFormat()
162         */
163        @Override
164        protected String nextSequenceValueFormat()
165        {
166                return "NEXTVAL(''{0}'')";
167        }
168
169        /**
170         * @see net.sf.hajdbc.dialect.StandardDialect#alterIdentityColumnFormat()
171         */
172        @Override
173        protected String alterIdentityColumnFormat()
174        {
175                return "ALTER SEQUENCE {0}_{1}_seq RESTART WITH {2}";
176        }
177
178        /**
179         * @see net.sf.hajdbc.dialect.StandardDialect#currentTimestampPattern()
180         */
181        @Override
182        protected String currentTimestampPattern()
183        {
184                return super.currentTimestampPattern() + "|(?<=\\W)NOW\\s*\\(\\s*\\)|(?<=\\W)TRANSACTION_TIMESTAMP\\s*\\(\\s*\\)|(?<=\\W)STATEMENT_TIMESTAMP\\s*\\(\\s*\\)|(?<=\\W)CLOCK_TIMESTAMP\\s*\\(\\s*\\)";
185        }
186
187        /**
188         * @see net.sf.hajdbc.dialect.StandardDialect#randomPattern()
189         */
190        @Override
191        protected String randomPattern()
192        {
193                return "(?<=\\W)RANDOM\\s*\\(\\s*\\)";
194        }
195
196        /**
197         * Recognizes FOR SHARE and FOR UPDATE.
198         * @see net.sf.hajdbc.dialect.StandardDialect#selectForUpdatePattern()
199         */
200        @Override
201        protected String selectForUpdatePattern()
202        {
203                return "SELECT\\s+.+\\s+FOR\\s+(SHARE|UPDATE)";
204        }
205
206        /**
207         * {@inheritDoc}
208         * @see net.sf.hajdbc.dialect.StandardDialect#getDumpRestoreSupport()
209         */
210        @Override
211        public DumpRestoreSupport getDumpRestoreSupport()
212        {
213                return this;
214        }
215
216        /**
217         * {@inheritDoc}
218         * @see net.sf.hajdbc.DumpRestoreSupport#createDumpProcess(net.sf.hajdbc.ConnectionProperties, java.io.File)
219         */
220        @Override
221        public ProcessBuilder createDumpProcess(ConnectionProperties properties, File file)
222        {
223                return this.setPassword(new ProcessBuilder("pg_dump", "-h", properties.getHost(), "-p", properties.getPort(), "-U", properties.getUser(), "-f", file.getPath(), "-F", "tar", properties.getDatabase()), properties);
224        }
225
226        /**
227         * {@inheritDoc}
228         * @see net.sf.hajdbc.DumpRestoreSupport#createRestoreProcess(net.sf.hajdbc.ConnectionProperties, java.io.File)
229         */
230        @Override
231        public ProcessBuilder createRestoreProcess(ConnectionProperties properties, File file)
232        {
233                return this.setPassword(new ProcessBuilder("pg_restore", "-h", properties.getHost(), "-p", properties.getPort(), "-U", properties.getUser(), "-d", properties.getDatabase(), file.getPath()), properties);
234        }
235        
236        private ProcessBuilder setPassword(ProcessBuilder builder, ConnectionProperties properties)
237        {
238                if (!PASSWORD_FILE.exists())
239                {
240                        builder.environment().put("PGPASSWORD", properties.getPassword());
241                }
242                
243                return builder;
244        }
245
246        /**
247         * {@inheritDoc}
248         * @see net.sf.hajdbc.dialect.StandardDialect#getTriggerSupport()
249         */
250        @Override
251        public TriggerSupport getTriggerSupport()
252        {
253                return this;
254        }
255
256        /**
257         * {@inheritDoc}
258         * @see net.sf.hajdbc.dialect.StandardDialect#createForeignKeyConstraintFormat()
259         */
260        @Override
261        protected String createTriggerFormat()
262        {
263                return "CREATE FUNCTION {0}_action() BEGIN {4} END; CREATE TRIGGER {0} {1} {2} ON {3} FOR EACH ROW EXECUTE PROCEDURE {0}_action()";
264        }
265
266        /**
267         * {@inheritDoc}
268         * @see net.sf.hajdbc.dialect.StandardDialect#dropTriggerFormat()
269         */
270        @Override
271        protected String dropTriggerFormat()
272        {
273                return "DROP TRIGGER {0} ON {1}; DROP FUNCTION {0}_action()";
274        }
