018package net.sf.hajdbc.dialect.mysql;
019
020import java.io.File;
021import java.sql.DatabaseMetaData;
022import java.sql.SQLException;
023import java.util.Collections;
024import java.util.List;
025
026import net.sf.hajdbc.ConnectionProperties;
027import net.sf.hajdbc.DumpRestoreSupport;
028import net.sf.hajdbc.dialect.StandardDialect;
029import net.sf.hajdbc.util.Strings;
030
031/**
032 * Dialect for <a href="http://www.mysql.com/products/database/mysql/">MySQL</a>
033 * @author Paul Ferraro
034 */
035@SuppressWarnings("nls")
036public class MySQLDialect extends StandardDialect implements DumpRestoreSupport
037{
038        private static final File PASSWORD_FILE = new File(String.format("%s%s.my.cnf", Strings.USER_HOME, Strings.FILE_SEPARATOR));
039        
040        /**
041         * {@inheritDoc}
042         * @see net.sf.hajdbc.dialect.StandardDialect#vendorPattern()
043         */
044        @Override
045        protected String vendorPattern()
046        {
047                return "mysql";
048        }
049
050        /**
051         * @see net.sf.hajdbc.dialect.StandardDialect#getDefaultSchemas(java.sql.DatabaseMetaData)
052         */
053        @Override
054        public List<String> getDefaultSchemas(DatabaseMetaData metaData) throws SQLException
055        {
056                return Collections.singletonList(this.executeFunction(metaData.getConnection(), "DATABASE()"));
057        }
058
059        /**
060         * Deferrability clause is not supported.
061         * @see net.sf.hajdbc.dialect.StandardDialect#createForeignKeyConstraintFormat()
062         */
063        @Override
064        protected String createForeignKeyConstraintFormat()
065        {
066                return "ALTER TABLE {1} ADD CONSTRAINT {0} FOREIGN KEY ({2}) REFERENCES {3} ({4}) ON DELETE {5,choice,0#CASCADE|1#RESTRICT|2#SET NULL|3#NO ACTION|4#SET DEFAULT} ON UPDATE {6,choice,0#CASCADE|1#RESTRICT|2#SET NULL|3#NO ACTION|4#SET DEFAULT}";
067        }
068
069        /**
070         * @see net.sf.hajdbc.dialect.StandardDialect#createUniqueConstraintFormat()
071         */
072        @Override
073        protected String createUniqueConstraintFormat()
074        {
075                return "ALTER TABLE {1} ADD UNIQUE {0} ({2})";
076        }
077
078        /**
079         * @see net.sf.hajdbc.dialect.StandardDialect#dropForeignKeyConstraintFormat()
080         */
081        @Override
082        protected String dropForeignKeyConstraintFormat()
083        {
084                return "ALTER TABLE {1} DROP FOREIGN KEY {0}";
085        }
086
087        /**
088         * @see net.sf.hajdbc.dialect.StandardDialect#dropUniqueConstraintFormat()
089         */
090        @Override
091        protected String dropUniqueConstraintFormat()
092        {
093                return "ALTER TABLE {1} DROP INDEX {0}";
094        }
095
096        /**
097         * @see net.sf.hajdbc.dialect.StandardDialect#alterIdentityColumnFormat()
098         */
099        @Override
100        protected String alterIdentityColumnFormat()
101        {
102                return "ALTER TABLE {0} AUTO_INCREMENT = {2}";
103        }
104
105        /**
106         * @see net.sf.hajdbc.dialect.StandardDialect#currentDatePattern()
107         */
108        @Override
109        protected String currentDatePattern()
110        {
111                return super.currentDatePattern() + "|(?<=\\W)CURDATE\\s*\\(\\s*\\)";
112        }
113
114        /**
115         * @see net.sf.hajdbc.dialect.StandardDialect#currentTimePattern()
116         */
117        @Override
118        protected String currentTimePattern()
119        {
120                return super.currentTimePattern() + "|(?<=\\W)CURTIME\\s*\\(\\s*\\)";
121        }
122
123        /**
124         * @see net.sf.hajdbc.dialect.StandardDialect#currentTimestampPattern()
125         */
126        @Override
127        protected String currentTimestampPattern()
128        {
129                return super.currentTimestampPattern() + "|(?<=\\W)NOW\\s*\\(\\s*\\)|(?<=\\W)SYSDATE\\s*\\(\\s*\\)";
130        }
131        
132        /**
133         * @see net.sf.hajdbc.dialect.StandardDialect#dateLiteralFormat()
134         */
135        @Override
136        protected String dateLiteralFormat()
137        {
138                return this.timestampLiteralFormat();
139        }
140
141        /**
142         * @see net.sf.hajdbc.dialect.StandardDialect#timeLiteralFormat()
143         */
144        @Override
145        protected String timeLiteralFormat()
146        {
147                return this.timestampLiteralFormat();
148        }
149
150        /**
151         * @see net.sf.hajdbc.dialect.StandardDialect#timestampLiteralFormat()
152         */
153        @Override
154        protected String timestampLiteralFormat()
155        {
156                return "''{0}''";
157        }
158
159        /**
160         * {@inheritDoc}
161         * @see net.sf.hajdbc.dialect.StandardDialect#getDumpRestoreSupport()
162         */
163        @Override
164        public DumpRestoreSupport getDumpRestoreSupport()
165        {
166                return this;
167        }
168
169        /**
170         * {@inheritDoc}
171         * @see net.sf.hajdbc.DumpRestoreSupport#createDumpProcess(net.sf.hajdbc.ConnectionProperties, java.io.File)
172         */
173        @Override
174        public ProcessBuilder createDumpProcess(ConnectionProperties properties, File file)
175        {
176                return this.setPassword(new ProcessBuilder("mysqldump", "-h", properties.getHost(), "-P", properties.getPort(), "-u", properties.getUser(), properties.getDatabase(), ">", file.getPath()), properties);
177        }
178
179        /**
180         * {@inheritDoc}
181         * @see net.sf.hajdbc.DumpRestoreSupport#createRestoreProcess(net.sf.hajdbc.ConnectionProperties, java.io.File)
182         */
183        @Override
184        public ProcessBuilder createRestoreProcess(ConnectionProperties properties, File file)
185        {
186                return this.setPassword(new ProcessBuilder("mysql", "-h", properties.getHost(), "-P", properties.getPort(), "-u", properties.getUser(), properties.getDatabase(), "<", file.getPath()), properties);
187        }
188        
189        private ProcessBuilder setPassword(ProcessBuilder builder, ConnectionProperties properties)
190        {
191                if (!PASSWORD_FILE.exists())
192                {
193                        builder.environment().put("MYSQL_PWD", properties.getPassword());
194                }
195                
196                return builder;
197        }
198}
