069        private static final File PASSWORD_FILE = new File(String.format("%s%s.my.cnf", Strings.USER_HOME, Strings.FILE_SEPARATOR));
070        
071        /**
072         * {@inheritDoc}
073         * @see net.sf.hajdbc.dialect.StandardDialect#vendorPattern()
074         */
075        @Override
076        protected String vendorPattern()
077        {
078                return "mysql";
079        }
080
081        @Override
082        protected Set<String> reservedIdentifiers(DatabaseMetaData metaData)
083        {
084                return new HashSet<String>(Arrays.asList(RESERVED_KEY_WORDS));
085        }
086
087        /**
088         * @see net.sf.hajdbc.dialect.StandardDialect#getDefaultSchemas(java.sql.DatabaseMetaData)
089         */
090        @Override
091        public List<String> getDefaultSchemas(DatabaseMetaData metaData) throws SQLException
092        {
093                return Collections.singletonList(this.executeFunction(metaData.getConnection(), "DATABASE()"));
094        }
095
096        /**
097         * Deferrability clause is not supported.
098         * @see net.sf.hajdbc.dialect.StandardDialect#createForeignKeyConstraintFormat()
099         */
100        @Override
101        protected String createForeignKeyConstraintFormat()
102        {
103                return "ALTER TABLE {1} ADD CONSTRAINT {0} FOREIGN KEY ({2}) REFERENCES {3} ({4}) ON DELETE {5,choice,0#CASCADE|1#RESTRICT|2#SET NULL|3#NO ACTION|4#SET DEFAULT} ON UPDATE {6,choice,0#CASCADE|1#RESTRICT|2#SET NULL|3#NO ACTION|4#SET DEFAULT}";
104        }
105
106        /**
107         * @see net.sf.hajdbc.dialect.StandardDialect#createUniqueConstraintFormat()
108         */
109        @Override
110        protected String createUniqueConstraintFormat()
111        {
112                return "ALTER TABLE {1} ADD UNIQUE {0} ({2})";
113        }
114
115        /**
116         * @see net.sf.hajdbc.dialect.StandardDialect#dropForeignKeyConstraintFormat()
117         */
118        @Override
119        protected String dropForeignKeyConstraintFormat()
120        {
121                return "ALTER TABLE {1} DROP FOREIGN KEY {0}";
122        }
123
124        /**
125         * @see net.sf.hajdbc.dialect.StandardDialect#dropUniqueConstraintFormat()
126         */
127        @Override
128        protected String dropUniqueConstraintFormat()
129        {
130                return "ALTER TABLE {1} DROP INDEX {0}";
131        }
132
133        /**
134         * @see net.sf.hajdbc.dialect.StandardDialect#alterIdentityColumnFormat()
135         */
136        @Override
137        protected String alterIdentityColumnFormat()
138        {
139                return "ALTER TABLE {0} AUTO_INCREMENT = {2}";
140        }
141
142        /**
143         * @see net.sf.hajdbc.dialect.StandardDialect#currentDatePattern()
144         */
145        @Override
146        protected String currentDatePattern()
147        {
148                return super.currentDatePattern() + "|(?<=\\W)CURDATE\\s*\\(\\s*\\)";
149        }
150
151        /**
152         * @see net.sf.hajdbc.dialect.StandardDialect#currentTimePattern()
153         */
154        @Override
155        protected String currentTimePattern()
156        {
157                return super.currentTimePattern() + "|(?<=\\W)CURTIME\\s*\\(\\s*\\)";
158        }
159
160        /**
161         * @see net.sf.hajdbc.dialect.StandardDialect#currentTimestampPattern()
162         */
163        @Override
164        protected String currentTimestampPattern()
165        {
166                return super.currentTimestampPattern() + "|(?<=\\W)NOW\\s*\\(\\s*\\)|(?<=\\W)SYSDATE\\s*\\(\\s*\\)";
167        }
168        
169        /**
170         * @see net.sf.hajdbc.dialect.StandardDialect#dateLiteralFormat()
171         */
172        @Override
173        protected String dateLiteralFormat()
174        {
175                return this.timestampLiteralFormat();
176        }
177
178        /**
179         * @see net.sf.hajdbc.dialect.StandardDialect#timeLiteralFormat()
180         */
181        @Override
182        protected String timeLiteralFormat()
183        {
184                return this.timestampLiteralFormat();
185        }
186
187        /**
188         * @see net.sf.hajdbc.dialect.StandardDialect#timestampLiteralFormat()
189         */
190        @Override
191        protected String timestampLiteralFormat()
192        {
193                return "''{0}''";
194        }
195
196        /**
197         * {@inheritDoc}
198         * @see net.sf.hajdbc.dialect.StandardDialect#getDumpRestoreSupport()
199         */
200        @Override
201        public DumpRestoreSupport getDumpRestoreSupport()
202        {
203                return this;
204        }
205
206        /**
207         * {@inheritDoc}
208         * @see net.sf.hajdbc.DumpRestoreSupport#createDumpProcess(net.sf.hajdbc.ConnectionProperties, java.io.File)
209         */
210        @Override
211        public ProcessBuilder createDumpProcess(ConnectionProperties properties, File file)
212        {
213                return setPassword(new ProcessBuilder("mysqldump", "-h", properties.getHost(), "-P", properties.getPort(), "-u", properties.getUser(), properties.getDatabase(), ">", file.getPath()), properties);
214        }
215
216        /**
217         * {@inheritDoc}
218         * @see net.sf.hajdbc.DumpRestoreSupport#createRestoreProcess(net.sf.hajdbc.ConnectionProperties, java.io.File)
219         */
220        @Override
221        public ProcessBuilder createRestoreProcess(ConnectionProperties properties, File file)
222        {
223                return setPassword(new ProcessBuilder("mysql", "-h", properties.getHost(), "-P", properties.getPort(), "-u", properties.getUser(), properties.getDatabase(), "<", file.getPath()), properties);
224        }
225        
226        private static ProcessBuilder setPassword(ProcessBuilder builder, ConnectionProperties properties)
227        {
228                if (!PASSWORD_FILE.exists())
229                {
230                        builder.environment().put("MYSQL_PWD", properties.getPassword());
231                }
232                
233                return builder;
234        }
