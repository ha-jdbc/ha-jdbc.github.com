018package net.sf.hajdbc.dialect;

020import java.sql.DatabaseMetaData;
021import java.sql.SQLException;
022import java.util.Arrays;
023import java.util.HashSet;
024import java.util.Set;
025import java.util.regex.Pattern;
026
027import net.sf.hajdbc.IdentifierNormalizer;
028import net.sf.hajdbc.util.Strings;
029
030public class StandardIdentifierNormalizer implements IdentifierNormalizer
031{
032        // As defined in SQL-92 specification: http://www.andrew.cmu.edu/user/shadow/sql/sql1992.txt

059        
060        private static final Pattern UPPER_CASE_PATTERN = Pattern.compile("[A-Z]");
061        private static final Pattern LOWER_CASE_PATTERN = Pattern.compile("[a-z]");
062        
063        private final Set<String> reservedIdentifierSet = new HashSet<String>(SQL_92_RESERVED_WORDS.length);
064        private final Pattern identifierPattern;
065        private final String quote;
066        private final boolean supportsMixedCaseIdentifiers;
067        private final boolean supportsMixedCaseQuotedIdentifiers;
068        private final boolean storesLowerCaseIdentifiers;
069        private final boolean storesLowerCaseQuotedIdentifiers;
070        private final boolean storesUpperCaseIdentifiers;
071        private final boolean storesUpperCaseQuotedIdentifiers;
072        
073        public StandardIdentifierNormalizer(DatabaseMetaData metaData, Pattern identifierPattern) throws SQLException
074        {
075                this.identifierPattern = identifierPattern;
076                this.quote = metaData.getIdentifierQuoteString();
077                this.supportsMixedCaseIdentifiers = metaData.supportsMixedCaseIdentifiers();
078                this.supportsMixedCaseQuotedIdentifiers = metaData.supportsMixedCaseQuotedIdentifiers();
079                this.storesLowerCaseIdentifiers = metaData.storesLowerCaseIdentifiers();
080                this.storesLowerCaseQuotedIdentifiers = metaData.storesLowerCaseQuotedIdentifiers();
081                this.storesUpperCaseIdentifiers = metaData.storesUpperCaseIdentifiers();
082                this.storesUpperCaseQuotedIdentifiers = metaData.storesUpperCaseQuotedIdentifiers();
083                
084                this.reservedIdentifierSet.addAll(Arrays.asList(SQL_92_RESERVED_WORDS));
085                
086                for (String word: metaData.getSQLKeywords().split(Strings.COMMA))
087                {
088                        this.reservedIdentifierSet.add(word.toUpperCase());
089                }
090        }
091
092        @Override
093        public String normalize(String identifier)
094        {
095                if (identifier == null) return null;
096                
097                int quoteLength = this.quote.length();
098                
099                boolean quoted = identifier.startsWith(this.quote) && identifier.endsWith(this.quote);
100                // Strip any existing quoting
101                String raw = quoted ? identifier.substring(quoteLength, identifier.length() - quoteLength) : identifier;
102                
103                // Quote reserved identifiers
104                boolean requiresQuoting = this.reservedIdentifierSet.contains(raw.toUpperCase());
105                
106                // Quote identifiers containing special characters
107                requiresQuoting |= !this.identifierPattern.matcher(raw).matches();
108                
109                // Quote mixed-case identifiers if detected and supported by DBMS
110                requiresQuoting |= quoted && !this.supportsMixedCaseIdentifiers && this.supportsMixedCaseQuotedIdentifiers && ((this.storesLowerCaseIdentifiers && !this.storesLowerCaseQuotedIdentifiers && UPPER_CASE_PATTERN.matcher(raw).find()) || (this.storesUpperCaseIdentifiers && !this.storesUpperCaseQuotedIdentifiers && LOWER_CASE_PATTERN.matcher(raw).find()));
111                
112                return requiresQuoting ? this.quote + this.normalizeCaseQuoted(raw) + this.quote : this.normalizeCase(raw);
113        }
114
115        private String normalizeCase(String identifier)
116        {
117                if (this.storesLowerCaseIdentifiers) return identifier.toLowerCase();
118                
119                if (this.storesUpperCaseIdentifiers) return identifier.toUpperCase();
120                
121                return identifier;
122        }
123
124        private String normalizeCaseQuoted(String identifier)
125        {
126                if (this.storesLowerCaseQuotedIdentifiers) return identifier.toLowerCase();
127                
128                if (this.storesUpperCaseQuotedIdentifiers) return identifier.toUpperCase();
129                
130                return identifier;
131        }
