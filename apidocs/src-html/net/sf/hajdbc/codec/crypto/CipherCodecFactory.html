018package net.sf.hajdbc.codec.crypto;
019
020import java.io.File;
021import java.io.FileInputStream;
022import java.io.IOException;
023import java.io.InputStream;
024import java.io.Serializable;
025import java.security.GeneralSecurityException;
026import java.security.KeyStore;
027import java.sql.SQLException;
028import java.text.MessageFormat;
029
030import net.sf.hajdbc.codec.Codec;
031import net.sf.hajdbc.codec.CodecFactory;
032import net.sf.hajdbc.util.Resources;
033import net.sf.hajdbc.util.Strings;
034import net.sf.hajdbc.util.SystemProperties;
035
036/**
037 * Used to decrypt configuration file passwords using a symmetric key stored in a keystore.
038 * Use the following command to generate Base-64 encoded encrypted passwords for use in your config file:<br/>
039 * <p><code>java -classpath ha-jdbc.jar net.sf.hajdbc.codec.crypto.CipherCodecFactory [password]</code></p>
040 * The following system properties can be used to customize the properties of the key and/or keystore:
041 * <table>
042 *      <tr>
043 *              <th>Property</th>
044 *              <th>Default</th>
045 *      </tr>
046 *      <tr>
047 *              <td>ha-jdbc.keystore.file</td>
048 *              <td>~/.keystore</td>
049 *      </tr>
050 *      <tr>
051 *              <td>ha-jdbc.keystore.type</td>
052 *              <td>jceks</td>
053 *      </tr>
054 *      <tr>
055 *              <td>ha-jdbc.keystore.password</td>
056 *              <td><em>none</em></td>
057 *      </tr>
058 *      <tr>
059 *              <td>ha-jdbc.key.alias</td>
060 *              <td>ha-jdbc</td>
061 *      </tr>
062 *      <tr>
063 *              <td>ha-jdbc.key.password</td>
064 *              <td><em>required</em><td>
065 *      </tr>
066 * </table>
067 * @author Paul Ferraro
068 */
069public class CipherCodecFactory implements CodecFactory, Serializable
070{
071        private static final long serialVersionUID = -4409167180573651279L;
072        
073        public static final String DEFAULT_KEYSTORE_FILE = String.format("%s/.keystore", SystemProperties.getSystemProperty("user.home"));
074        public static final String DEFAULT_KEY_ALIAS = "ha-jdbc";
075        
076        enum Property
077        {
078                KEYSTORE_FILE("keystore.file", DEFAULT_KEYSTORE_FILE),
079                KEYSTORE_TYPE("keystore.type", "jceks"),
080                KEYSTORE_PASSWORD("keystore.password", null),
081                KEY_ALIAS("key.alias", DEFAULT_KEY_ALIAS),
082                KEY_PASSWORD("key.password", null);
083                
084                final String nameFormat;
085                final String name;
086                final String defaultValue;
087                
088                private Property(String name, String defaultValue)
089                {
090                        this.nameFormat = "ha-jdbc.{0}." + name;
091                        this.name = "ha-jdbc." + name;
092                        this.defaultValue = defaultValue;
093                }
094        }
095        
096        private String getProperty(String id, Property property)
097        {
098                String value = SystemProperties.getSystemProperty(MessageFormat.format(property.nameFormat, id));
099                
100                if (value != null) return value;
101                
102                String pattern = SystemProperties.getSystemProperty(property.name, property.defaultValue);
103                
104                if (pattern == null) return null;
105                
106                return MessageFormat.format(pattern, id);
107        }
108        
109        @Override
110        public String getId()
111        {
112                return "?";
113        }
114
115        /**
116         * {@inheritDoc}
117         * @see net.sf.hajdbc.codec.CodecFactory#createCodec(java.lang.String)
118         */
119        @Override
120        public Codec createCodec(String clusterId) throws SQLException
121        {
122                String type = this.getProperty(clusterId, Property.KEYSTORE_TYPE);
123                File file = new File(this.getProperty(clusterId, Property.KEYSTORE_FILE));
124                String password = this.getProperty(clusterId, Property.KEYSTORE_PASSWORD);
125
126                String keyAlias = this.getProperty(clusterId, Property.KEY_ALIAS);
127                String keyPassword = this.getProperty(clusterId, Property.KEY_PASSWORD);
128                
129                try
130                {
131                        KeyStore store = KeyStore.getInstance(type);
132                        
133                        InputStream input = new FileInputStream(file);
134                        try
135                        {
136                                store.load(input, (password != null) ? password.toCharArray() : null);
137                        }
138                        finally
139                        {
140                                Resources.close(input);
141                        }
142                        return new CipherCodec(store.getKey(keyAlias, (keyPassword != null) ? keyPassword.toCharArray() : null));
143                }
144                catch (GeneralSecurityException e)
145                {
146                        throw new SQLException(e);
147                }
148                catch (IOException e)
149                {
150                        throw new SQLException(e);
151                }
152        }
153        
154        public static void main(String... args)
155        {
156                if (args.length != 2)
157                {
158                        System.err.println(String.format("Usage:%s\tjava %s <cluster-id> <password-to-encrypt>", Strings.NEW_LINE, CipherCodecFactory.class.getName()));
159                        System.exit(1);
160                        return;
161                }
162                
163                String clusterId = args[0];
164                String value = args[1];
165                
166                try
167                {
168                        Codec codec = new CipherCodecFactory().createCodec(clusterId);
169
170                        System.out.println(codec.encode(value));
171                }
172                catch (SQLException e)
173                {
174                        e.printStackTrace(System.err);
175                }
176        }
177}
