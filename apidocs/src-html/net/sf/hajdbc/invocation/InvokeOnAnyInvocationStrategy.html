018package net.sf.hajdbc.invocation;
019
020import java.util.Map;
021import java.util.SortedMap;
022
023import net.sf.hajdbc.Database;
024import net.sf.hajdbc.DatabaseCluster;
025import net.sf.hajdbc.ExceptionFactory;
026import net.sf.hajdbc.Messages;
027import net.sf.hajdbc.dialect.Dialect;
028import net.sf.hajdbc.logging.Level;
029import net.sf.hajdbc.logging.Logger;
030import net.sf.hajdbc.logging.LoggerFactory;
031import net.sf.hajdbc.sql.SQLProxy;
032import net.sf.hajdbc.state.StateManager;
033import net.sf.hajdbc.util.Collections;
034
035/**
036 * @author Paul Ferraro
037 */
038public class InvokeOnAnyInvocationStrategy extends InvokeOnNextInvocationStrategy
039{
040        private static Logger logger = LoggerFactory.getLogger(InvokeOnExistingInvocationStrategy.class);
041
042        /**
043         * {@inheritDoc}
044         * @see net.sf.hajdbc.invocation.InvocationStrategy#invoke(net.sf.hajdbc.sql.SQLProxy, net.sf.hajdbc.invocation.Invoker)
045         */
046        @Override
047        public <Z, D extends Database<Z>, T, R, E extends Exception> SortedMap<D, R> invoke(SQLProxy<Z, D, T, E> proxy, Invoker<Z, D, T, R, E> invoker) throws E
048        {
049                DatabaseCluster<Z, D> cluster = proxy.getDatabaseCluster();
050                Dialect dialect = cluster.getDialect();
051                StateManager stateManager = cluster.getStateManager();
052                ExceptionFactory<E> exceptionFactory = proxy.getExceptionFactory();
053
054                for (Map.Entry<D, T> entry: proxy.entries())
055                {
056                        D database = entry.getKey();
057                        
058                        try
059                        {
060                                R result = invoker.invoke(database, entry.getValue());
061                                
062                                return Collections.singletonSortedMap(database, result);
063                        }
064                        catch (Exception e)
065                        {
066                                E exception = exceptionFactory.createException(e);
067                                
068                                if (exceptionFactory.indicatesFailure(exception, dialect) && (cluster.getBalancer().size() > 1))
069                                {
070                                        if (cluster.deactivate(database, stateManager))
071                                        {
072                                                logger.log(Level.ERROR, exception, Messages.DATABASE_DEACTIVATED.getMessage(), database, cluster);
073                                        }
074                                }
075                                else
076                                {
077                                        throw exception;
078                                }
079                        }
080                }
081                
082                return super.invoke(proxy, invoker);
083        }
084}
