018package net.sf.hajdbc.invocation;
019
020import java.util.ArrayList;
021import java.util.List;
022import java.util.Map;
023import java.util.Set;
024import java.util.SortedMap;
025import java.util.TreeMap;
026import java.util.AbstractMap;
027import java.util.concurrent.Callable;
028import java.util.concurrent.ExecutionException;
029import java.util.concurrent.ExecutorService;
030import java.util.concurrent.Future;
031
032import net.sf.hajdbc.Database;
033import net.sf.hajdbc.DatabaseCluster;
034import net.sf.hajdbc.ExceptionFactory;
035import net.sf.hajdbc.Messages;
036import net.sf.hajdbc.sql.ProxyFactory;
037
038/**
039 * @author Paul Ferraro
040 */
041public class AllResultsCollector implements InvokeOnManyInvocationStrategy.ResultsCollector
042{
043        public static interface ExecutorProvider
044        {
045                <Z, D extends Database<Z>> ExecutorService getExecutor(DatabaseCluster<Z, D> cluster);
046        }
047        
048        private final ExecutorProvider provider;
049        
050        public AllResultsCollector(ExecutorProvider provider)
051        {
052                this.provider = provider;
053        }
054        
055        /**
056         * {@inheritDoc}
057         * @see net.sf.hajdbc.invocation.InvokeOnManyInvocationStrategy#collectResults(net.sf.hajdbc.sql.SQLProxy, net.sf.hajdbc.invocation.Invoker)
058         */
059        @Override
060        public <Z, D extends Database<Z>, T, R, E extends Exception> Map.Entry<SortedMap<D, R>, SortedMap<D, E>> collectResults(ProxyFactory<Z, D, T, E> map, final Invoker<Z, D, T, R, E> invoker)
061        {
062                DatabaseCluster<Z, D> cluster = map.getDatabaseCluster();
063                ExceptionFactory<E> exceptionFactory = map.getExceptionFactory();
064                Set<D> databaseSet = cluster.getBalancer();
065                
066                map.getRoot().retain(databaseSet);
067                
068                if (databaseSet.isEmpty())
069                {
070                        exceptionFactory.createException(Messages.NO_ACTIVE_DATABASES.getMessage(cluster));
071                }
072
073                int size = databaseSet.size();
074                
075                List<Invocation<Z, D, T, R, E>> invocationList = new ArrayList<Invocation<Z, D, T, R, E>>(size);
076                
077                for (D database: databaseSet)
078                {
079                        invocationList.add(new Invocation<Z, D, T, R, E>(invoker, database, map.get(database)));
080                }
081                
082                try
083                {
084                        List<Future<R>> futureList = this.provider.getExecutor(cluster).invokeAll(invocationList);
085                        
086                        final SortedMap<D, R> resultMap = new TreeMap<D, R>();
087                        final SortedMap<D, E> exceptionMap = new TreeMap<D, E>();
088                        
089                        for (int i = 0; i < invocationList.size(); ++i)
090                        {
091                                D database = invocationList.get(i).getDatabase();
092                                
093                                try
094                                {
095                                        resultMap.put(database, futureList.get(i).get());
096                                }
097                                catch (ExecutionException e)
098                                {
099                                        exceptionMap.put(database, exceptionFactory.createException(e.getCause()));
100                                }
101                                catch (InterruptedException e)
102                                {
103                                        Thread.currentThread().interrupt();
104                                        
105                                        exceptionMap.put(database, exceptionFactory.createException(e));
106                                }
107                        }
108                
109                        return new AbstractMap.SimpleImmutableEntry<SortedMap<D, R>, SortedMap<D, E>>(resultMap, exceptionMap);
110                }
111                catch (InterruptedException e)
112                {
113                        throw new IllegalStateException(e);
114                }
115        }
116        
117        private static class Invocation<Z, D extends Database<Z>, T, R, E extends Exception> implements Callable<R>
118        {
119                private final Invoker<Z, D, T, R, E> invoker;
120                private final D database;
121                private final T object;
122                
123                Invocation(Invoker<Z, D, T, R, E> invoker, D database, T object)
124                {
125                        this.invoker = invoker;
126                        this.database = database;
127                        this.object = object;
128                }
129                
130                D getDatabase()
131                {
132                        return this.database;
133                }
134                
135                /**
136                 * {@inheritDoc}
137                 * @see java.util.concurrent.Callable#call()
138                 */
139                @Override
140                public R call() throws E
141                {
142                        return this.invoker.invoke(this.database, this.object);
143                }
144        }
145}
