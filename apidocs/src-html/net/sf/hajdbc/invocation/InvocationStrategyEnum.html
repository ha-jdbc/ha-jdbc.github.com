001package net.sf.hajdbc.invocation;
002
003import java.util.SortedMap;
004
005import net.sf.hajdbc.Database;
006import net.sf.hajdbc.sql.SQLProxy;
007
008public enum InvocationStrategyEnum implements InvocationStrategy
009{
010        INVOKE_ON_ALL(new InvokeOnAllInvocationStrategy()),
011        INVOKE_ON_ANY(new InvokeOnAnyInvocationStrategy()),
012        INVOKE_ON_EXISTING(new InvokeOnExistingInvocationStrategy()),
013        INVOKE_ON_NEXT(new InvokeOnNextInvocationStrategy()),
014        INVOKE_ON_PRIMARY(new InvokeOnPrimaryInvocationStrategy()),
015        TRANSACTION_INVOKE_ON_ALL(new TransactionInvokeOnAllInvocationStrategy(false)),
016        END_TRANSACTION_INVOKE_ON_ALL(new TransactionInvokeOnAllInvocationStrategy(true)),
017        ;
018
019        private final InvocationStrategy strategy;
020        
021        private InvocationStrategyEnum(InvocationStrategy strategy)
022        {
023                this.strategy = strategy;
024        }
025        
026        @Override
027        public <Z, D extends Database<Z>, T, R, E extends Exception> SortedMap<D, R> invoke(SQLProxy<Z, D, T, E> proxy, Invoker<Z, D, T, R, E> invoker) throws E
028        {
029                return this.strategy.invoke(proxy, invoker);
030        }
031}
