001package net.sf.hajdbc.invocation;
002
003import java.util.SortedMap;
004
005import net.sf.hajdbc.Database;
006import net.sf.hajdbc.sql.SQLProxy;
007
008public interface InvocationStrategy
009{
010        /**
011         * Invoke the specified invoker against the specified proxy.
012         * @param proxy a JDBC object proxy
013         * @param invoker an invoker
014         * @return a result
015         * @throws Exception if the invocation fails
016         */
017        <Z, D extends Database<Z>, T, R, E extends Exception> SortedMap<D, R> invoke(SQLProxy<Z, D, T, E> proxy, Invoker<Z, D, T, R, E> invoker) throws E;
018}
