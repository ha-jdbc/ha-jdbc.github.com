018package net.sf.hajdbc.sql;
019
020import javax.naming.BinaryRefAddr;
021import javax.naming.Reference;
022import javax.naming.StringRefAddr;
023import javax.naming.spi.ObjectFactory;
024
025import net.sf.hajdbc.Database;
026import net.sf.hajdbc.DatabaseClusterConfiguration;
027import net.sf.hajdbc.DatabaseClusterConfigurationFactory;
028import net.sf.hajdbc.util.Objects;
029import net.sf.hajdbc.xml.XMLDatabaseClusterConfigurationFactory;
030
031/**
032 * @author Paul Ferraro
033 * @param <Z> 
034 */
035public abstract class CommonDataSourceReference<Z> extends Reference
036{
037        private static final long serialVersionUID = 1408239702660701511L;
038        
039        protected static final String CLUSTER = "cluster"; //$NON-NLS-1$
040        protected static final String CONFIG = "config"; //$NON-NLS-1$
041        
042        /**
043         * Constructs a new CommonDataSourceReference
044         * @param <D> the datasource type
045         * @param targetClass the target class of the DataSource.
046         * @param factoryClass the ObjectFactory class for creating a DataSource
047         * @param cluster a database cluster
048         * @param configurationClass
049         * @param config the uri of the configuration file
050         */
051        protected <D extends Database<Z>> CommonDataSourceReference(Class<Z> targetClass, Class<? extends ObjectFactory> factoryClass, String cluster, Class<? extends DatabaseClusterConfiguration<Z, D>> configurationClass, String config)
052        {
053                this(targetClass, factoryClass, cluster, new XMLDatabaseClusterConfigurationFactory<Z, D>(configurationClass, cluster, config));
054        }
055        
056        /**
057         * Constructs a new CommonDataSourceReference
058         * @param <D> the datasource type
059         * @param targetClass the target class of the DataSource.
060         * @param factoryClass the ObjectFactory class for creating a DataSource
061         * @param cluster a database cluster
062         * @param factory a database cluster configuration factory
063         */
064        protected <D extends Database<Z>> CommonDataSourceReference(Class<Z> targetClass, Class<? extends ObjectFactory> factoryClass, String cluster, DatabaseClusterConfigurationFactory<Z, D> factory)
065        {
066                super(targetClass.getName(), factoryClass.getName(), null);
067                
068                if (cluster == null)
069                {
070                        throw new IllegalArgumentException();
071                }
072                
073                this.add(new StringRefAddr(CLUSTER, cluster));
074                
075                this.add(new BinaryRefAddr(CONFIG, Objects.serialize(factory)));
076        }
077}
