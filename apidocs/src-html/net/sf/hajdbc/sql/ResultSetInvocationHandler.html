048/**
049 * @author Paul Ferraro
050 * @param <D> 
051 * @param <S> 
052 */
053@SuppressWarnings("nls")
054public class ResultSetInvocationHandler<Z, D extends Database<Z>, S extends Statement> extends ChildInvocationHandler<Z, D, S, ResultSet, SQLException>
055{
056        private static final Set<Method> driverReadMethodSet = Methods.findMethods(ResultSet.class, "findColumn", "getConcurrency", "getCursorName", "getFetchDirection", "getFetchSize", "getHoldability", "getMetaData", "getRow", "getType", "getWarnings", "isAfterLast", "isBeforeFirst", "isClosed", "isFirst", "isLast", "row(Deleted|Inserted|Updated)", "wasNull");
057        private static final Set<Method> driverWriteMethodSet = Methods.findMethods(ResultSet.class, "absolute", "afterLast", "beforeFirst", "cancelRowUpdates", "clearWarnings", "first", "last", "moveTo(Current|Insert)Row", "next", "previous", "relative", "setFetchDirection", "setFetchSize");
058        private static final Set<Method> transactionalWriteMethodSet = Methods.findMethods(ResultSet.class, "(delete|insert|update)Row");
059        private static final Set<Method> getBlobMethodSet = Methods.findMethods(ResultSet.class, "getBlob");
060        private static final Set<Method> getClobMethodSet = Methods.findMethods(ResultSet.class, "getClob");
061        private static final Set<Method> getNClobMethodSet = Methods.findMethods(ResultSet.class, "getNClob");
062        private static final Set<Method> getSQLXMLMethodSet = Methods.findMethods(ResultSet.class, "getSQLXML");
063        
064        private static final Method closeMethod = Methods.getMethod(ResultSet.class, "close");
065        private static final Method getStatementMethod = Methods.getMethod(ResultSet.class, "getStatement");
066        
067        protected FileSupport<SQLException> fileSupport;
068        private TransactionContext<Z, D> transactionContext;
069        private List<Invoker<Z, D, ResultSet, ?, SQLException>> invokerList = new LinkedList<Invoker<Z, D, ResultSet, ?, SQLException>>();
070        
071        /**
072         * @param statement the statement that created this result set
073         * @param proxy the invocation handler of the statement that created this result set
074         * @param invoker the invoker that was used to create this result set
075         * @param resultSetMap a map of database to underlying result set
076         * @param transactionContext 
077         * @param fileSupport support for streams
078         * @throws Exception
079         */
080        protected ResultSetInvocationHandler(S statement, SQLProxy<Z, D, S, SQLException> proxy, Invoker<Z, D, S, ResultSet, SQLException> invoker, Map<D, ResultSet> resultSetMap, TransactionContext<Z, D> transactionContext, FileSupport<SQLException> fileSupport)
081        {
082                super(statement, proxy, invoker, ResultSet.class, SQLException.class, resultSetMap);
083                
084                this.transactionContext = transactionContext;
085                this.fileSupport = fileSupport;
086        }
087
088        /**
089         * {@inheritDoc}
090         * @see net.sf.hajdbc.sql.AbstractChildInvocationHandler#getParentMethod()
091         */
092        @Override
093        protected Method getParentMethod()
094        {
095                return getStatementMethod;
096        }
097
098        /**
099         * {@inheritDoc}
100         * @see net.sf.hajdbc.sql.AbstractInvocationHandler#getInvocationHandlerFactory(java.lang.Object, java.lang.reflect.Method, java.lang.Object[])
101         */
102        @Override
103        protected InvocationHandlerFactory<Z, D, ResultSet, ?, SQLException> getInvocationHandlerFactory(ResultSet object, Method method, Object[] parameters) throws SQLException
104        {
105                if (getBlobMethodSet.contains(method))
106                {
107                        return new BlobInvocationHandlerFactory<Z, D, ResultSet>(this.getParent().getConnection());
108                }
109                
110                if (getClobMethodSet.contains(method))
111                {
112                        return new ClobInvocationHandlerFactory<Z, D, ResultSet>(this.getParent().getConnection());
113                }
114                
115                if (getNClobMethodSet.contains(method))
116                {
117                        return new NClobInvocationHandlerFactory<Z, D, ResultSet>(this.getParent().getConnection());
118                }
119                
120                if (getSQLXMLMethodSet.contains(method))
121                {
122                        return new SQLXMLInvocationHandlerFactory<Z, D, ResultSet>(this.getParent().getConnection());
123                }
124                
125                return super.getInvocationHandlerFactory(object, method, parameters);
126        }
127
128        /**
129         * @see net.sf.hajdbc.sql.AbstractChildInvocationHandler#getInvocationStrategy(java.lang.Object, java.lang.reflect.Method, java.lang.Object[])
130         */
131        @Override
132        protected InvocationStrategy getInvocationStrategy(ResultSet resultSet, Method method, Object[] parameters) throws SQLException
133        {
134                if (driverReadMethodSet.contains(method))
135                {
136                        return InvocationStrategyEnum.INVOKE_ON_ANY;
137                }
138                
139                if (driverWriteMethodSet.contains(method) || method.equals(closeMethod))
140                {
141                        return InvocationStrategyEnum.INVOKE_ON_EXISTING;
142                }
143                
144                if (transactionalWriteMethodSet.contains(method))
145                {
146                        return this.transactionContext.start(InvocationStrategyEnum.TRANSACTION_INVOKE_ON_ALL, this.getParent().getConnection());
147                }
148                
149                if (this.isGetMethod(method))
150                {
151                        return InvocationStrategyEnum.INVOKE_ON_ANY;
152                }
153                
154                if (this.isUpdateMethod(method))
155                {
156                        return InvocationStrategyEnum.INVOKE_ON_EXISTING;
157                }
158                
159                return super.getInvocationStrategy(resultSet, method, parameters);
160        }
161
162        /**
163         * @see net.sf.hajdbc.sql.AbstractChildInvocationHandler#getInvoker(java.lang.Object, java.lang.reflect.Method, java.lang.Object[])
164         */
165        @Override
166        protected <R> Invoker<Z, D, ResultSet, R, SQLException> getInvoker(ResultSet object, final Method method, final Object[] parameters) throws SQLException
167        {
168                if (this.isUpdateMethod(method) && (parameters.length > 1))
169                {
170                        Object typeParameter = parameters[1];
171                        
172                        if (typeParameter != null)
173                        {
174                                Class<?> type = method.getParameterTypes()[1];
175                                
176                                if (type.equals(InputStream.class))
177                                {
178                                        final File file = this.fileSupport.createFile((InputStream) typeParameter);
179                                        
180                                        return new Invoker<Z, D, ResultSet, R, SQLException>()
181                                        {
182                                                @Override
183                                                public R invoke(D database, ResultSet resultSet) throws SQLException
184                                                {
185                                                        List<Object> parameterList = new ArrayList<Object>(Arrays.asList(parameters));
186                                                        
187                                                        parameterList.set(1, ResultSetInvocationHandler.this.fileSupport.getInputStream(file));
188                                                        
189                                                        return Methods.<R, SQLException>invoke(method, ResultSetInvocationHandler.this.getExceptionFactory(), resultSet, parameterList.toArray());
190                                                }                               
191                                        };
192                                }
193                                
194                                if (type.equals(Reader.class))
195                                {
196                                        final File file = this.fileSupport.createFile((Reader) typeParameter);
197                                        
198                                        return new Invoker<Z, D, ResultSet, R, SQLException>()
199                                        {
200                                                @Override
201                                                public R invoke(D database, ResultSet resultSet) throws SQLException
202                                                {
203                                                        List<Object> parameterList = new ArrayList<Object>(Arrays.asList(parameters));
204                                                        
205                                                        parameterList.set(1, ResultSetInvocationHandler.this.fileSupport.getReader(file));
206                                                        
207                                                        return Methods.<R, SQLException>invoke(method, ResultSetInvocationHandler.this.getExceptionFactory(), resultSet, parameterList.toArray());
208                                                }                               
209                                        };
210                                }
211                                
212                                if (type.equals(Blob.class))
213                                {
214                                        Blob blob = (Blob) typeParameter;
215                                        
216                                        if (Proxy.isProxyClass(blob.getClass()) && (Proxy.getInvocationHandler(blob) instanceof SQLProxy<?, ?, ?, ?>))
217                                        {
218                                                final SQLProxy<Z, D, Blob, SQLException> proxy = this.getInvocationHandler(blob);
219                                                
220                                                return new Invoker<Z, D, ResultSet, R, SQLException>()
221                                                {
222                                                        @Override
223                                                        public R invoke(D database, ResultSet resultSet) throws SQLException
224                                                        {
225                                                                List<Object> parameterList = new ArrayList<Object>(Arrays.asList(parameters));
226                                                                
227                                                                parameterList.set(1, proxy.getObject(database));
228                                                                
229                                                                return Methods.<R, SQLException>invoke(method, ResultSetInvocationHandler.this.getExceptionFactory(), resultSet, parameterList.toArray());
230                                                        }                               
231                                                };
232                                        }
233
234                                        parameters[1] = new SerialBlob(blob);
235                                }
236                                
237                                // Handle both clob and nclob
238                                if (Clob.class.isAssignableFrom(type))
239                                {
240                                        Clob clob = (Clob) typeParameter;
241                                        
242                                        if (Proxy.isProxyClass(clob.getClass()) && (Proxy.getInvocationHandler(clob) instanceof SQLProxy<?, ?, ?, ?>))
243                                        {
244                                                final SQLProxy<Z, D, Clob, SQLException> proxy = this.getInvocationHandler(clob);
245                                                
246                                                return new Invoker<Z, D, ResultSet, R, SQLException>()
247                                                {
248                                                        @Override
249                                                        public R invoke(D database, ResultSet resultSet) throws SQLException
250                                                        {
251                                                                List<Object> parameterList = new ArrayList<Object>(Arrays.asList(parameters));
252                                                                
253                                                                parameterList.set(1, proxy.getObject(database));
254                                                                
255                                                                return Methods.<R, SQLException>invoke(method, ResultSetInvocationHandler.this.getExceptionFactory(), resultSet, parameterList.toArray());
256                                                        }                               
257                                                };
258                                        }
259
260                                        Clob serialClob = new SerialClob(clob);
261                                        
262                                        parameters[1] = type.equals(Clob.class) ? serialClob : ProxyFactory.createProxy(type, new SimpleInvocationHandler(serialClob));
263                                }
264                        }
265                }
266                
267                return super.getInvoker(object, method, parameters);
268        }

270        /**
271         * @see net.sf.hajdbc.sql.AbstractChildInvocationHandler#postInvoke(java.lang.Object, java.lang.reflect.Method, java.lang.Object[])
272         */
273        @Override
274        protected void postInvoke(ResultSet object, Method method, Object[] parameters)
275        {
276                if (method.equals(closeMethod))
277                {
278                        this.getParentProxy().removeChild(this);
279                }
280        }
281
282        /**
283         * @see net.sf.hajdbc.sql.AbstractChildInvocationHandler#close(java.lang.Object, java.lang.Object)
284         */
285        @Override
286        protected void close(S statement, ResultSet resultSet) throws SQLException
287        {
288                resultSet.close();
289        }
290
291        /**
292         * {@inheritDoc}
293         * @see net.sf.hajdbc.sql.AbstractInvocationHandler#record(net.sf.hajdbc.invocation.Invoker, java.lang.reflect.Method, java.lang.Object[])
294         */
295        @Override
296        protected void record(Invoker<Z, D, ResultSet, ?, SQLException> invoker, Method method, Object[] parameters)
297        {
298                if (driverWriteMethodSet.contains(method) || this.isUpdateMethod(method))
299                {
300                        synchronized (this.invokerList)
301                        {
302                                this.invokerList.add(invoker);
303                        }
304                }
305                else
306                {
307                        super.record(invoker, method, parameters);
308                }
309        }
310
311        /**
312         * @see net.sf.hajdbc.sql.AbstractInvocationHandler#replay(net.sf.hajdbc.Database, java.lang.Object)
313         */
314        @Override
315        protected void replay(D database, ResultSet resultSet) throws SQLException
316        {
317                super.replay(database, resultSet);
318                
319                synchronized (this.invokerList)
320                {
321                        for (Invoker<Z, D, ResultSet, ?, SQLException> invoker: this.invokerList)
322                        {
323                                invoker.invoke(database, resultSet);
324                        }
325                }
326        }
327        
328        private boolean isGetMethod(Method method)
329        {
330                Class<?>[] types = method.getParameterTypes();
331                
332                return method.getName().startsWith("get") && (types != null) && (types.length > 0) && (types[0].equals(String.class) || types[0].equals(Integer.TYPE));
333        }
334        
335        private boolean isUpdateMethod(Method method)
336        {
337                Class<?>[] types = method.getParameterTypes();
338                
339                return method.getName().startsWith("update") && (types != null) && (types.length > 0) && (types[0].equals(String.class) || types[0].equals(Integer.TYPE));
340        }
