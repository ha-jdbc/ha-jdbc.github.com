032import java.util.Arrays;

034import java.util.List;
035import java.util.Map;
036import java.util.Set;
037import java.util.concurrent.locks.Lock;
038
039import javax.sql.rowset.serial.SerialBlob;
040import javax.sql.rowset.serial.SerialClob;
041
042import net.sf.hajdbc.Database;
043import net.sf.hajdbc.invocation.InvocationStrategy;
044import net.sf.hajdbc.invocation.InvocationStrategyEnum;
045import net.sf.hajdbc.invocation.Invoker;
046import net.sf.hajdbc.util.reflect.Methods;
047import net.sf.hajdbc.util.reflect.ProxyFactory;
048import net.sf.hajdbc.util.reflect.SimpleInvocationHandler;

050/**
051 * @author Paul Ferraro
052 * @param <D> 
053 * @param <S> 
054 */
055@SuppressWarnings("nls")
056public abstract class AbstractPreparedStatementInvocationHandler<Z, D extends Database<Z>, S extends PreparedStatement> extends AbstractStatementInvocationHandler<Z, D, S>
057{
058        private static final Set<Method> databaseReadMethodSet = Methods.findMethods(PreparedStatement.class, "getMetaData", "getParameterMetaData");
059        private static final Method executeMethod = Methods.getMethod(PreparedStatement.class, "execute");
060        private static final Method executeUpdateMethod = Methods.getMethod(PreparedStatement.class, "executeUpdate");
061        private static final Method executeQueryMethod = Methods.getMethod(PreparedStatement.class, "executeQuery");
062        private static final Method clearParametersMethod = Methods.getMethod(PreparedStatement.class, "clearParameters");
063        private static final Method addBatchMethod = Methods.getMethod(PreparedStatement.class, "addBatch");
064        
065        protected List<Lock> lockList = Collections.emptyList();
066        protected boolean selectForUpdate = false;
067        private final Set<Method> setMethodSet;
068        
069        /**
070         * @param connection
071         * @param proxy
072         * @param invoker
073         * @param statementClass 
074         * @param statementMap
075         * @param transactionContext 
076         * @param fileSupport 
077         * @throws Exception
078         */
079        protected AbstractPreparedStatementInvocationHandler(Connection connection, SQLProxy<Z, D, Connection, SQLException> proxy, Invoker<Z, D, Connection, S, SQLException> invoker, Class<S> statementClass, Map<D, S> statementMap, TransactionContext<Z, D> transactionContext, FileSupport<SQLException> fileSupport, Set<Method> setMethodSet)
080        {
081                super(connection, proxy, invoker, statementClass, statementMap, transactionContext, fileSupport);
082                
083                this.setMethodSet = setMethodSet;
084        }
085        
086        /**
087         * {@inheritDoc}
088         * @see net.sf.hajdbc.sql.AbstractStatementInvocationHandler#getInvocationHandlerFactory(java.sql.Statement, java.lang.reflect.Method, java.lang.Object[])
089         */
090        @Override
091        protected InvocationHandlerFactory<Z, D, S, ?, SQLException> getInvocationHandlerFactory(S object, Method method, Object[] parameters) throws SQLException
092        {
093                if (method.equals(executeQueryMethod))
094                {
095                        return new ResultSetInvocationHandlerFactory<Z, D, S>(this.transactionContext, this.fileSupport);
096                }
097                
098                return super.getInvocationHandlerFactory(object, method, parameters);
099        }
100
101        /**
102         * @see net.sf.hajdbc.sql.AbstractStatementInvocationHandler#getInvocationStrategy(java.sql.Statement, java.lang.reflect.Method, java.lang.Object[])
103         */
104        @Override
105        protected InvocationStrategy getInvocationStrategy(S statement, Method method, Object[] parameters) throws SQLException
106        {
107                if (databaseReadMethodSet.contains(method))
108                {
109                        return InvocationStrategyEnum.INVOKE_ON_NEXT;
110                }
111                
112                if (this.setMethodSet.contains(method) || method.equals(clearParametersMethod) || method.equals(addBatchMethod))
113                {
114                        return InvocationStrategyEnum.INVOKE_ON_EXISTING;
115                }
116                
117                if (method.equals(executeMethod) || method.equals(executeUpdateMethod))
118                {
119                        return this.transactionContext.start(new LockingInvocationStrategy(InvocationStrategyEnum.TRANSACTION_INVOKE_ON_ALL, this.lockList), this.getParent());
120                }
121                
122                if (method.equals(executeQueryMethod))
123                {
124                        int concurrency = statement.getResultSetConcurrency();
125                        
126                        if (this.lockList.isEmpty() && (concurrency == ResultSet.CONCUR_READ_ONLY) && !this.selectForUpdate)
127                        {
128                                return InvocationStrategyEnum.INVOKE_ON_NEXT;
129                        }
130                        
131                        InvocationStrategy strategy = new LockingInvocationStrategy(InvocationStrategyEnum.TRANSACTION_INVOKE_ON_ALL, this.lockList);
132                        
133                        return this.selectForUpdate ? this.transactionContext.start(strategy, this.getParent()) : strategy;
134                }
135                
136                return super.getInvocationStrategy(statement, method, parameters);
137        }
138
139        /**
140         * @see net.sf.hajdbc.sql.AbstractChildInvocationHandler#getInvoker(java.lang.Object, java.lang.reflect.Method, java.lang.Object[])
141         */
142        @Override
143        protected <R> Invoker<Z, D, S, R, SQLException> getInvoker(S statement, final Method method, final Object[] parameters) throws SQLException
144        {
145                if (this.isParameterSetMethod(method) && (parameters.length > 1))
146                {
147                        Object typeParameter = parameters[1];
148                        
149                        if (typeParameter != null)
150                        {
151                                Class<?> type = method.getParameterTypes()[1];
152                                
153                                if (type.equals(InputStream.class))
154                                {
155                                        final File file = this.fileSupport.createFile((InputStream) typeParameter);
156                                        
157                                        return new Invoker<Z, D, S, R, SQLException>()
158                                        {
159                                                @Override
160                                                public R invoke(D database, S statement) throws SQLException
161                                                {
162                                                        List<Object> parameterList = new ArrayList<Object>(Arrays.asList(parameters));
163                                                        
164                                                        parameterList.set(1, AbstractPreparedStatementInvocationHandler.this.fileSupport.getInputStream(file));
165                                                        
166                                                        return Methods.<R, SQLException>invoke(method, AbstractPreparedStatementInvocationHandler.this.getExceptionFactory(), statement, parameterList.toArray());
167                                                }                               
168                                        };
169                                }
170                                
171                                if (type.equals(Reader.class))
172                                {
173                                        final File file = this.fileSupport.createFile((Reader) typeParameter);
174                                        
175                                        return new Invoker<Z, D, S, R, SQLException>()
176                                        {
177                                                @Override
178                                                public R invoke(D database, S statement) throws SQLException
179                                                {
180                                                        List<Object> parameterList = new ArrayList<Object>(Arrays.asList(parameters));
181                                                        
182                                                        parameterList.set(1, AbstractPreparedStatementInvocationHandler.this.fileSupport.getReader(file));
183                                                        
184                                                        return Methods.<R, SQLException>invoke(method, AbstractPreparedStatementInvocationHandler.this.getExceptionFactory(), statement, parameterList.toArray());
185                                                }                               
186                                        };
187                                }
188                                
189                                if (type.equals(Blob.class))
190                                {
191                                        Blob blob = (Blob) typeParameter;
192                                        
193                                        if (Proxy.isProxyClass(blob.getClass()) && (Proxy.getInvocationHandler(blob) instanceof SQLProxy<?, ?, ?, ?>))
194                                        {
195                                                final SQLProxy<Z, D, Blob, SQLException> proxy = this.getInvocationHandler(blob);
196                                                
197                                                return new Invoker<Z, D, S, R, SQLException>()
198                                                {
199                                                        @Override
200                                                        public R invoke(D database, S statement) throws SQLException
201                                                        {
202                                                                List<Object> parameterList = new ArrayList<Object>(Arrays.asList(parameters));
203                                                                
204                                                                parameterList.set(1, proxy.getObject(database));
205                                                                
206                                                                return Methods.<R, SQLException>invoke(method, AbstractPreparedStatementInvocationHandler.this.getExceptionFactory(), statement, parameterList.toArray());
207                                                        }                               
208                                                };
209                                        }
210
211                                        parameters[1] = new SerialBlob(blob);
212                                }
213                                
214                                // Handle both clob and nclob
215                                if (Clob.class.isAssignableFrom(type))
216                                {
217                                        Clob clob = (Clob) typeParameter;
218                                        
219                                        if (Proxy.isProxyClass(clob.getClass()) && (Proxy.getInvocationHandler(clob) instanceof SQLProxy<?, ?, ?, ?>))
220                                        {
221                                                final SQLProxy<Z, D, Clob, SQLException> proxy = this.getInvocationHandler(clob);
222                                                
223                                                return new Invoker<Z, D, S, R, SQLException>()
224                                                {
225                                                        @Override
226                                                        public R invoke(D database, S statement) throws SQLException
227                                                        {
228                                                                List<Object> parameterList = new ArrayList<Object>(Arrays.asList(parameters));
229                                                                
230                                                                parameterList.set(1, proxy.getObject(database));
231                                                                
232                                                                return Methods.<R, SQLException>invoke(method, AbstractPreparedStatementInvocationHandler.this.getExceptionFactory(), statement, parameterList.toArray());
233                                                        }                               
234                                                };
235                                        }
236
237                                        Clob serialClob = new SerialClob(clob);
238                                        
239                                        parameters[1] = type.equals(Clob.class) ? serialClob : ProxyFactory.createProxy(type, new SimpleInvocationHandler(serialClob));
240                                }
241                        }
242                }
243                
244                return super.getInvoker(statement, method, parameters);
245        }
246        
247        /**
248         * @see net.sf.hajdbc.sql.AbstractStatementInvocationHandler#isBatchMethod(java.lang.reflect.Method)
249         */
250        @Override
251        protected boolean isBatchMethod(Method method)
252        {
253                return method.equals(addBatchMethod) || method.equals(clearParametersMethod) || this.isParameterSetMethod(method) || super.isBatchMethod(method);
254        }
255
256        private boolean isParameterSetMethod(Method method)
257        {
258                Class<?>[] types = method.getParameterTypes();
259                
260                return this.setMethodSet.contains(method) && (types.length > 0) && this.isIndexType(types[0]);
261        }
262        
263        protected boolean isIndexType(Class<?> type)
264        {
265                return type.equals(Integer.TYPE);
266        }
