018package net.sf.hajdbc.sql;
019
020import java.lang.reflect.Method;
021import java.lang.reflect.Proxy;
022import java.sql.Connection;
023import java.sql.SQLException;
024import java.sql.Savepoint;
025import java.util.Arrays;
026import java.util.HashSet;
027import java.util.Map;
028import java.util.Set;
029
030import net.sf.hajdbc.Database;
031import net.sf.hajdbc.durability.Durability;
032import net.sf.hajdbc.invocation.InvocationStrategy;
033import net.sf.hajdbc.invocation.InvocationStrategyEnum;
034import net.sf.hajdbc.invocation.Invoker;
035import net.sf.hajdbc.util.StaticRegistry;
036import net.sf.hajdbc.util.reflect.Methods;
037
038/**
039 * @author Paul Ferraro
040 * @param <D> 
041 * @param <P> 
042 */
043@SuppressWarnings("nls")
044public class ConnectionInvocationHandler<Z, D extends Database<Z>, P> extends ChildInvocationHandler<Z, D, P, Connection, SQLException>
045{
046        private static final Set<Method> driverReadMethodSet = Methods.findMethods(Connection.class, "create(ArrayOf|Struct)", "getAutoCommit", "getCatalog", "getClientInfo", "getHoldability", "getTypeMap", "getWarnings", "isClosed", "isReadOnly", "nativeSQL");
047        private static final Set<Method> databaseReadMethodSet = Methods.findMethods(Connection.class, "getTransactionIsolation", "isValid");
048        private static final Set<Method> driverWriterMethodSet = Methods.findMethods(Connection.class, "clearWarnings", "setClientInfo", "setHoldability", "setTypeMap");
049        private static final Set<Method> createStatementMethodSet = Methods.findMethods(Connection.class, "createStatement");
050        private static final Set<Method> prepareStatementMethodSet = Methods.findMethods(Connection.class, "prepareStatement");
051        private static final Set<Method> prepareCallMethodSet = Methods.findMethods(Connection.class, "prepareCall");
052        private static final Set<Method> setSavepointMethodSet = Methods.findMethods(Connection.class, "setSavepoint");
053
054        private static final Method setAutoCommitMethod = Methods.getMethod(Connection.class, "setAutoCommit", Boolean.TYPE);
055        private static final Method commitMethod = Methods.getMethod(Connection.class, "commit");
056        private static final Method rollbackMethod = Methods.getMethod(Connection.class, "rollback");
057        private static final Method getMetaDataMethod = Methods.getMethod(Connection.class, "getMetaData");
058        private static final Method releaseSavepointMethod = Methods.getMethod(Connection.class, "releaseSavepoint", Savepoint.class);
059        private static final Method rollbackSavepointMethod = Methods.getMethod(Connection.class, "rollback", Savepoint.class);
060        private static final Method closeMethod = Methods.getMethod(Connection.class, "close");
061        private static final Method createBlobMethod = Methods.getMethod(Connection.class, "createBlob");
062        private static final Method createClobMethod = Methods.getMethod(Connection.class, "createClob");
063        private static final Method createNClobMethod = Methods.getMethod(Connection.class, "createNClob");
064        private static final Method createSQLXMLMethod = Methods.getMethod(Connection.class, "createSQLXML");
065        
066        private static final Set<Method> endTransactionMethodSet = new HashSet<Method>(Arrays.asList(commitMethod, rollbackMethod, setAutoCommitMethod));
067        private static final Set<Method> createLocatorMethodSet = new HashSet<Method>(Arrays.asList(createBlobMethod, createClobMethod, createNClobMethod, createSQLXMLMethod));
068        
069        private static final StaticRegistry<Method, Durability.Phase> phaseRegistry = new DurabilityPhaseRegistry(Arrays.asList(commitMethod, setAutoCommitMethod), Arrays.asList(rollbackMethod));
070        
071        private TransactionContext<Z, D> transactionContext;
072        
073        /**
074         * Constructs a new ConnectionInvocationHandler
075         * @param proxy
076         * @param handler
077         * @param invoker
078         * @param connectionMap
079         * @param transactionContext
080         */
081        public ConnectionInvocationHandler(P proxy, SQLProxy<Z, D, P, SQLException> handler, Invoker<Z, D, P, Connection, SQLException> invoker, Map<D, Connection> connectionMap, TransactionContext<Z, D> transactionContext)
082        {
083                super(proxy, handler, invoker, Connection.class, SQLException.class, connectionMap);
084                
085                this.transactionContext = transactionContext;
086        }
087        
088        /**
089         * {@inheritDoc}
090         * @see net.sf.hajdbc.sql.AbstractInvocationHandler#getInvocationHandlerFactory(java.lang.Object, java.lang.reflect.Method, java.lang.Object[])
091         */
092        @Override
093        protected InvocationHandlerFactory<Z, D, Connection, ?, SQLException> getInvocationHandlerFactory(Connection connection, Method method, Object[] parameters)
094        {
095                if (createStatementMethodSet.contains(method))
096                {
097                        return new StatementInvocationHandlerFactory<Z, D>(this.transactionContext);
098                }
099                
100                if (prepareStatementMethodSet.contains(method))
101                {
102                        return new PreparedStatementInvocationHandlerFactory<Z, D>(this.transactionContext, (String) parameters[0]);
103                }
104                
105                if (prepareCallMethodSet.contains(method))
106                {
107                        return new CallableStatementInvocationHandlerFactory<Z, D>(this.transactionContext);
108                }
109                
110                if (setSavepointMethodSet.contains(method))
111                {
112                        return new SavepointInvocationHandlerFactory<Z, D>();
113                }
114                
115                if (method.equals(getMetaDataMethod))
116                {
117                        return new DatabaseMetaDataInvocationHandlerFactory<Z, D>();
118                }
119                
120                if (method.equals(createBlobMethod))
121                {
122                        return new BlobInvocationHandlerFactory<Z, D, Connection>(connection);
123                }
124
125                if (method.equals(createClobMethod))
126                {
127                        return new ClobInvocationHandlerFactory<Z, D, Connection>(connection);
128                }
129
130                if (method.equals(createNClobMethod))
131                {
132                        return new NClobInvocationHandlerFactory<Z, D, Connection>(connection);
133                }
134                
135                if (method.equals(createSQLXMLMethod))
136                {
137                        return new SQLXMLInvocationHandlerFactory<Z, D, Connection>(connection);
138                }
139                
140                return null;
141        }
142
143        /**
144         * @throws SQLException 
145         * @see net.sf.hajdbc.sql.AbstractChildInvocationHandler#getInvocationStrategy(java.lang.Object, java.lang.reflect.Method, java.lang.Object[])
146         */
147        @Override
148        protected InvocationStrategy getInvocationStrategy(Connection connection, Method method, Object[] parameters) throws SQLException
149        {
150                if (driverReadMethodSet.contains(method))
151                {
152                        return InvocationStrategyEnum.INVOKE_ON_ANY;
153                }
154                
155                if (databaseReadMethodSet.contains(method) || method.equals(getMetaDataMethod))
156                {
157                        return InvocationStrategyEnum.INVOKE_ON_NEXT;
158                }
159                
160                if (driverWriterMethodSet.contains(method) || method.equals(closeMethod) || createStatementMethodSet.contains(method))
161                {
162                        return InvocationStrategyEnum.INVOKE_ON_EXISTING;
163                }
164                
165                if (prepareStatementMethodSet.contains(method) || prepareCallMethodSet.contains(method) || createLocatorMethodSet.contains(method))
166                {
167                        return InvocationStrategyEnum.INVOKE_ON_ALL;
168                }
169                
170                if (endTransactionMethodSet.contains(method))
171                {
172                        return this.transactionContext.end(InvocationStrategyEnum.END_TRANSACTION_INVOKE_ON_ALL, phaseRegistry.get(method));
173                }
174                
175                if (method.equals(rollbackSavepointMethod) || method.equals(releaseSavepointMethod))
176                {
177                        return InvocationStrategyEnum.END_TRANSACTION_INVOKE_ON_ALL;
178                }
179                
180                if (setSavepointMethodSet.contains(method))
181                {
182                        return InvocationStrategyEnum.TRANSACTION_INVOKE_ON_ALL;
183                }
184                
185                return super.getInvocationStrategy(connection, method, parameters);
186        }
187
188        /**
189         * @see net.sf.hajdbc.sql.AbstractChildInvocationHandler#getInvoker(java.lang.Object, java.lang.reflect.Method, java.lang.Object[])
190         */
191        @Override
192        protected <R> Invoker<Z, D, Connection, R, SQLException> getInvoker(Connection connection, Method method, Object[] parameters) throws SQLException
193        {
194                if (method.equals(releaseSavepointMethod))
195                {
196                        final SQLProxy<Z, D, Savepoint, SQLException> proxy = this.getInvocationHandler((Savepoint) parameters[0]);
197                        
198                        return new Invoker<Z, D, Connection, R, SQLException>()
199                        {
200                                @Override
201                                public R invoke(D database, Connection connection) throws SQLException
202                                {
203                                        connection.releaseSavepoint(proxy.getObject(database));
204                                        
205                                        return null;
206                                }                                       
207                        };
208                }
209                
210                if (method.equals(rollbackSavepointMethod))
211                {
212                        final SQLProxy<Z, D, Savepoint, SQLException> proxy = this.getInvocationHandler((Savepoint) parameters[0]);
213                        
214                        return new Invoker<Z, D, Connection, R, SQLException>()
215                        {
216                                @Override
217                                public R invoke(D database, Connection connection) throws SQLException
218                                {
219                                        connection.rollback(proxy.getObject(database));
220                                        
221                                        return null;
222                                }                                       
223                        };
224                }
225                
226                Invoker<Z, D, Connection, R, SQLException> invoker = super.getInvoker(connection, method, parameters);
227                
228                if (endTransactionMethodSet.contains(method))
229                {
230                        return this.transactionContext.end(invoker, phaseRegistry.get(method));
231                }
232                
233                return invoker;
234        }
235
236        /**
237         * @see net.sf.hajdbc.sql.AbstractChildInvocationHandler#isSQLMethod(java.lang.reflect.Method)
238         */
239        @Override
240        protected boolean isSQLMethod(Method method)
241        {
242                return prepareStatementMethodSet.contains(method);
243        }
244
245        @Override
246        protected boolean isRecordable(Method method)
247        {
248                return driverWriterMethodSet.contains(method) || method.equals(setAutoCommitMethod);
249        }
250
251        /**
252         * @see net.sf.hajdbc.sql.AbstractChildInvocationHandler#postInvoke(java.lang.Object, java.lang.reflect.Method, java.lang.Object[])
253         */
254        @Override
255        protected void postInvoke(Connection object, Method method, Object[] parameters)
256        {
257                if (method.equals(closeMethod))
258                {
259                        this.transactionContext.close();
260                        
261                        this.getParentProxy().removeChild(this);
262                }
263                else if (method.equals(releaseSavepointMethod))
264                {
265                        @SuppressWarnings("unchecked")
266                        SQLProxy<Z, D, Savepoint, SQLException> proxy = (SQLProxy<Z, D, Savepoint, SQLException>) Proxy.getInvocationHandler(parameters[0]);
267                        
268                        this.removeChild(proxy);
269                }
270        }
271
272        /**
273         * @see net.sf.hajdbc.sql.AbstractChildInvocationHandler#close(java.lang.Object, java.lang.Object)
274         */
275        @Override
276        protected void close(P parent, Connection connection) throws SQLException
277        {
278                connection.close();
279        }
