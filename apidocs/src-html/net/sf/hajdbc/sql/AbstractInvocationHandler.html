057@SuppressWarnings("nls")
058public abstract class AbstractInvocationHandler<Z, D extends Database<Z>, T, E extends Exception> implements InvocationHandler, SQLProxy<Z, D, T, E>
059{
060        private static final Method equalsMethod = Methods.getMethod(Object.class, "equals", Object.class);
061        private static final Method hashCodeMethod = Methods.getMethod(Object.class, "hashCode");
062        private static final Method toStringMethod = Methods.getMethod(Object.class, "toString");
063        private static final Set<Method> wrapperMethods = Methods.findMethods(Wrapper.class, "isWrapperFor", "unwrap");
064        
065        protected Logger logger = LoggerFactory.getLogger(this.getClass());
066        
067        private final Class<T> proxyClass;
068        private final Map<D, T> objectMap;
069        private final Map<SQLProxy<Z, D, ?, ? extends Exception>, Void> childMap = new WeakHashMap<SQLProxy<Z, D, ?, ? extends Exception>, Void>();
070        private final Map<Method, Invoker<Z, D, T, ?, E>> invokerMap = new HashMap<Method, Invoker<Z, D, T, ?, E>>();
071        private final Class<E> exceptionClass;

073        /**
074         * Constructs a new AbstractInvocationHandler
075         * @param proxyClass the interface being proxied
076         * @param exceptionClass the class for exceptions thrown by this object
077         * @param objectMap a map of database to sql object.
078         */
079        protected AbstractInvocationHandler(Class<T> proxyClass, Class<E> exceptionClass, Map<D, T> objectMap)
080        {
081                this.proxyClass = proxyClass;
082                this.exceptionClass = exceptionClass;
083                this.objectMap = objectMap;
084        }
085        
086        /**
087         * @see java.lang.reflect.InvocationHandler#invoke(java.lang.Object, java.lang.reflect.Method, java.lang.Object[])
088         */
089        @Override
090        public Object invoke(Object object, Method method, Object[] parameters) throws Throwable
091        {
092                DatabaseCluster<Z, D> cluster = this.getDatabaseCluster();
093                
094                if (!cluster.isActive())
095                {
096                        throw new SQLException(Messages.CLUSTER_NOT_ACTIVE.getMessage(cluster));
097                }
098                
099                return this.invokeOnProxy(this.proxyClass.cast(object), method, parameters);
100        }
101
102        private <R> R invokeOnProxy(T object, Method method, Object[] parameters) throws E
103        {
104                InvocationStrategy strategy = this.getInvocationStrategy(object, method, parameters);
105
106                Invoker<Z, D, T, R, E> invoker = this.getInvoker(object, method, parameters);
107                
108                this.logger.log(Level.TRACE, "Invoking {0} using {1}", method, strategy.getClass().getName());
109                
110                SortedMap<D, R> results = strategy.invoke(this, invoker);
111                
112                this.record(invoker, method, parameters);
113                
114                this.postInvoke(object, method, parameters);
115                
116                @SuppressWarnings("unchecked")
117                InvocationHandlerFactory<Z, D, T, R, E> handlerFactory = (InvocationHandlerFactory<Z, D, T, R, E>) this.getInvocationHandlerFactory(object, method, parameters);
118                
119                InvocationResultFactory<Z, D, R, E> resultFactory = (handlerFactory != null) ? new ProxyInvocationResultFactory<R>(handlerFactory, object, invoker) : new SimpleInvocationResultFactory<R>();
120                
121                return resultFactory.createResult(results);
122        }
123        
124        protected InvocationHandlerFactory<Z, D, T, ?, E> getInvocationHandlerFactory(T object, Method method, Object[] parameters) throws E
125        {
126                return null;
127        }
128        
129        /**
130         * Returns the appropriate {@link InvocationStrategy} for the specified method.
131         * This implementation detects {@link java.sql.Wrapper} methods; and {@link Object#equals}, {@link Object#hashCode()}, and {@link Object#toString()}.
132         * Default invocation strategy is {@link InvokeOnAllInvocationStrategy}. 
133         * @param object the proxied object
134         * @param method the method to invoke
135         * @param parameters the method invocation parameters
136         * @return an invocation strategy
137         * @throws Exception
138         */
139        protected InvocationStrategy getInvocationStrategy(T object, Method method, Object[] parameters) throws E
140        {
141                if (method.equals(equalsMethod) || method.equals(hashCodeMethod) || method.equals(toStringMethod) || wrapperMethods.contains(method))
142                {
143                        return InvocationStrategyEnum.INVOKE_ON_ANY;
144                }
145
146                return InvocationStrategyEnum.INVOKE_ON_ALL;
147        }
148        
149        /**
150         * Return the appropriate invoker for the specified method.
151         * @param object
152         * @param method
153         * @param parameters
154         * @return an invoker
155         * @throws Exception
156         */
157        protected <R> Invoker<Z, D, T, R, E> getInvoker(T object, Method method, Object[] parameters) throws E
158        {
159                if (this.isSQLMethod(method))
160                {
161                        List<Object> parameterList = new ArrayList<Object>(Arrays.asList(parameters));
162                        
163                        long now = System.currentTimeMillis();
164                        
165                        DatabaseCluster<Z, D> cluster = this.getDatabaseCluster();
166                        Dialect dialect = cluster.getDialect();
167                        
168                        if (cluster.isCurrentTimestampEvaluationEnabled())
169                        {
170                                parameterList.set(0, dialect.evaluateCurrentTimestamp((String) parameterList.get(0), new java.sql.Timestamp(now)));
171                        }
172                        
173                        if (cluster.isCurrentDateEvaluationEnabled())
174                        {
175                                parameterList.set(0, dialect.evaluateCurrentDate((String) parameterList.get(0), new java.sql.Date(now)));
176                        }
177                        
178                        if (cluster.isCurrentTimeEvaluationEnabled())
179                        {
180                                parameterList.set(0, dialect.evaluateCurrentTime((String) parameterList.get(0), new java.sql.Time(now)));
181                        }
182                        
183                        if (cluster.isRandEvaluationEnabled())
184                        {
185                                parameterList.set(0, dialect.evaluateRand((String) parameterList.get(0)));
186                        }
187                        
188                        return new SimpleInvoker<R>(method, parameterList.toArray());
189                }
190                
191                return new SimpleInvoker<R>(method, parameters);
192        }

265        /**
266         * Returns the underlying SQL object for the specified database.
267         * If the sql object does not exist (this might be the case if the database was newly activated), it will be created from the stored operation.
268         * Any recorded operations are also executed. If the object could not be created, or if any of the executed operations failed, then the specified database is deactivated.
269         * @param database a database descriptor.
270         * @return an underlying SQL object
271         */
272        @Override
273        public T getObject(D database)
274        {
275                synchronized (this.objectMap)
276                {
277                        T object = this.objectMap.get(database);
278                        
279                        if (object == null)
280                        {
281                                try
282                                {
283                                        object = this.createObject(database);
284                                        
285                                        this.replay(database, object);
286                                        
287                                        this.objectMap.put(database, object);
288                                }
289                                catch (Throwable e)
290                                {
291                                        DatabaseCluster<Z, D> cluster = this.getDatabaseCluster();
292                                        
293                                        if (!this.objectMap.isEmpty() && cluster.deactivate(database, cluster.getStateManager()))
294                                        {
295                                                this.logger.log(Level.WARN, e, Messages.SQL_OBJECT_INIT_FAILED.getMessage(), this.getClass().getName(), database);
296                                        }
297                                }
298                        }
299                        
300                        return object;
301                }
302        }
303        
304        protected abstract T createObject(D database) throws E;
305
306        protected void record(Invoker<Z, D, T, ?, E> invoker, Method method, Object[] parameters)
307        {
308                // Record only the last invocation of a given set*(...) method
309                if (this.isRecordable(method))
310                {
311                        synchronized (this.invokerMap)
312                        {
313                                this.invokerMap.put(method, invoker);
314                        }
315                }
316        }
317        
318        protected boolean isRecordable(Method method)
319        {
320                return false;
321        }
322        
323        protected void replay(D database, T object) throws E
324        {
325                synchronized (this.invokerMap)
326                {
327                        for (Invoker<Z, D, T, ?, E> invoker: this.invokerMap.values())
328                        {
329                                this.logger.log(Level.TRACE, "Replaying against database {0}: {1}.{2}", database, object.getClass().getName(), invoker);
330
331                                invoker.invoke(database, object);
332                        }
333                }
334        }
335        
336        /**
337         * @see net.sf.hajdbc.sql.SQLProxy#retain(java.util.Set)
338         */
339        @Override
340        public final void retain(Set<D> databaseSet)
341        {
342                synchronized (this.childMap)
343                {
344                        for (SQLProxy<Z, D, ?, ? extends Exception> child: this.childMap.keySet())
345                        {
346                                child.retain(databaseSet);
347                        }
348                }
349                
350                synchronized (this.objectMap)
351                {
352                        Iterator<Map.Entry<D, T>> mapEntries = this.objectMap.entrySet().iterator();
353                        
354                        while (mapEntries.hasNext())
355                        {
356                                Map.Entry<D, T> mapEntry = mapEntries.next();
357                                
358                                D database = mapEntry.getKey();
359                                
360                                if (!databaseSet.contains(database))
361                                {
362                                        T object = mapEntry.getValue();
363                                        
364                                        if (object != null)
365                                        {
366                                                this.close(database, object);
367                                        }
368                                        
369                                        mapEntries.remove();
370                                }
371                        }
372                }
373        }

393        protected class SimpleInvoker<R> implements Invoker<Z, D, T, R, E>
394        {
395                private final Method method;
396                private final Object[] parameters;
397                
398                /**
399                 * @param method
400                 * @param parameters
401                 */
402                public SimpleInvoker(Method method, Object[] parameters)
403                {
404                        this.method = method;
405                        this.parameters = parameters;
406                }
407                
408                /**
409                 * @see net.sf.hajdbc.invocation.Invoker#invoke(net.sf.hajdbc.Database, java.lang.Object)
410                 */
411                @Override
412                public R invoke(D database, T object) throws E
413                {
414                        return Methods.<R, E>invoke(this.method, AbstractInvocationHandler.this.getExceptionFactory(), object, this.parameters);
415                }
416
417                /**
418                 * {@inheritDoc}
419                 * @see java.lang.Object#toString()
420                 */
421                @Override
422                public String toString()
423                {
424                        return this.method.toString();
425                }
426        }

428        class SimpleInvocationResultFactory<R> implements InvocationResultFactory<Z, D, R, E>
429        {
430                @Override
431                public R createResult(SortedMap<D, R> resultMap)
432                {
433                        assert !resultMap.isEmpty();
434                        
435                        DatabaseCluster<Z, D> cluster = AbstractInvocationHandler.this.getDatabaseCluster();
436                        Iterator<Map.Entry<D, R>> results = resultMap.entrySet().iterator();
437
438                        R primaryResult = results.next().getValue();
439
440                        while (results.hasNext())
441                        {
442                                Map.Entry<D, R> entry = results.next();
443                                R result = entry.getValue();
444                                
445                                if (!Objects.equals(primaryResult, result))
446                                {
447                                        D database = entry.getKey();
448                                        
449                                        if (cluster.deactivate(database, cluster.getStateManager()))
450                                        {
451                                                AbstractInvocationHandler.this.logger.log(Level.ERROR, Messages.DATABASE_INCONSISTENT.getMessage(), database, cluster, primaryResult, result);
452                                        }
453                                }
454                        }
455                        
456                        return primaryResult;
457                }
458        }

460        class ProxyInvocationResultFactory<R> implements InvocationResultFactory<Z, D, R, E>
461        {
462                private final InvocationHandlerFactory<Z, D, T, R, E> factory;
463                private final T object;
464                private final Invoker<Z, D, T, R, E> invoker;
465                
466                ProxyInvocationResultFactory(InvocationHandlerFactory<Z, D, T, R, E> factory, T object, Invoker<Z, D, T, R, E> invoker)
467                {
468                        this.factory = factory;
469                        this.object = object;
470                        this.invoker = invoker;
471                }
472                
473                @Override
474                public R createResult(SortedMap<D, R> results) throws E
475                {
476                        return ProxyFactory.createProxy(this.factory.getTargetClass(), this.factory.createInvocationHandler(this.object, AbstractInvocationHandler.this, this.invoker, results));
477                }
478        }
