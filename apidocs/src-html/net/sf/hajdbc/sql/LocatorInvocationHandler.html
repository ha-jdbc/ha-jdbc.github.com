018package net.sf.hajdbc.sql;
019
020import java.lang.reflect.Method;
021import java.sql.SQLException;
022import java.util.LinkedList;
023import java.util.List;
024import java.util.Map;
025import java.util.Set;
026
027import net.sf.hajdbc.Database;
028import net.sf.hajdbc.invocation.InvocationStrategy;
029import net.sf.hajdbc.invocation.InvocationStrategyEnum;
030import net.sf.hajdbc.invocation.Invoker;
031import net.sf.hajdbc.logging.Level;
032import net.sf.hajdbc.util.reflect.Methods;
033
034/**
035 * @author Paul Ferraro
036 * @param <Z>
037 * @param <D>
038 * @param <P>
039 * @param <T>
040 */
041public abstract class LocatorInvocationHandler<Z, D extends Database<Z>, P, T> extends ChildInvocationHandler<Z, D, P, T, SQLException>
042{
043        private final Method freeMethod;
044        private final Set<Method> readMethodSet;
045        private final Set<Method> writeMethodSet;
046        private final List<Invoker<Z, D, T, ?, SQLException>> invokerList = new LinkedList<Invoker<Z, D, T, ?, SQLException>>();
047        private final boolean updateCopy;
048        
049        /**
050         * Constructs a new LocatorInvocationHandler
051         * @param parent
052         * @param proxy
053         * @param invoker
054         * @param locatorClass
055         * @param locators
056         * @param updateCopy
057         * @param readMethodSet
058         * @param writeMethodSet
059         */
060        protected LocatorInvocationHandler(P parent, SQLProxy<Z, D, P, SQLException> proxy, Invoker<Z, D, P, T, SQLException> invoker, Class<T> locatorClass, Map<D, T> locators, boolean updateCopy, Set<Method> readMethodSet, Set<Method> writeMethodSet)
061        {
062                super(parent, proxy, invoker, locatorClass, SQLException.class, locators);
063                
064                this.freeMethod = Methods.findMethod(locatorClass, "free");
065                this.updateCopy = updateCopy;
066                this.readMethodSet = readMethodSet;
067                this.writeMethodSet = writeMethodSet;
068        }
069
070        /**
071         * @see net.sf.hajdbc.sql.AbstractChildInvocationHandler#getInvocationStrategy(java.lang.Object, java.lang.reflect.Method, java.lang.Object[])
072         */
073        @Override
074        protected InvocationStrategy getInvocationStrategy(T locator, Method method, Object[] parameters) throws SQLException
075        {
076                if (this.readMethodSet.contains(method))
077                {
078                        return this.updateCopy ? InvocationStrategyEnum.INVOKE_ON_ANY : InvocationStrategyEnum.INVOKE_ON_NEXT;
079                }
080                
081                if (this.updateCopy && this.writeMethodSet.contains(method))
082                {
083                        return InvocationStrategyEnum.INVOKE_ON_EXISTING;
084                }
085                
086                return super.getInvocationStrategy(locator, method, parameters);
087        }
088        
089        /**
090         * @see net.sf.hajdbc.sql.AbstractChildInvocationHandler#postInvoke(java.lang.Object, java.lang.reflect.Method, java.lang.Object[])
091         */
092        @Override
093        protected void postInvoke(T object, Method method, Object[] parameters)
094        {
095                if ((this.freeMethod != null) && method.equals(this.freeMethod))
096                {
097                        this.getParentProxy().removeChild(this);
098                }
099        }
100
101        /**
102         * @see net.sf.hajdbc.sql.AbstractChildInvocationHandler#close(java.lang.Object, java.lang.Object)
103         */
104        @SuppressWarnings("nls")
105        @Override
106        protected void close(P parent, T locator)
107        {
108                if (this.freeMethod != null)
109                {
110                        try
111                        {
112                                this.free(locator);
113                        }
114                        catch (SQLException e)
115                        {
116                                this.logger.log(Level.WARN, e, e.getMessage());
117                        }
118                }
119        }
120        
121        protected abstract void free(T locator) throws SQLException;
122
123        /**
124         * @see net.sf.hajdbc.sql.AbstractInvocationHandler#record(net.sf.hajdbc.invocation.Invoker, java.lang.reflect.Method, java.lang.Object[])
125         */
126        @Override
127        protected void record(Invoker<Z, D, T, ?, SQLException> invoker, Method method, Object[] parameters)
128        {
129                if (this.isRecordable(method))
130                {
131                        synchronized (this.invokerList)
132                        {
133                                this.invokerList.add(invoker);
134                        }
135                }
136        }
137}
