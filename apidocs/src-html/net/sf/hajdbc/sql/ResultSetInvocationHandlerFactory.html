018package net.sf.hajdbc.sql;
019
020import java.lang.reflect.InvocationHandler;
021import java.sql.ResultSet;
022import java.sql.SQLException;
023import java.sql.Statement;
024import java.util.Map;
025
026import net.sf.hajdbc.Database;
027import net.sf.hajdbc.invocation.Invoker;
028
029/**
030 * @author paul
031 *
032 */
033public class ResultSetInvocationHandlerFactory<Z, D extends Database<Z>, S extends Statement> implements InvocationHandlerFactory<Z, D, S, ResultSet, SQLException>
034{
035        private final FileSupport<SQLException> fileSupport;
036        private final TransactionContext<Z, D> transactionContext;
037
038        public ResultSetInvocationHandlerFactory(TransactionContext<Z, D> transactionContext, FileSupport<SQLException> fileSupport)
039        {
040                this.fileSupport = fileSupport;
041                this.transactionContext = transactionContext;
042        }
043
044        /**
045         * {@inheritDoc}
046         * @see net.sf.hajdbc.sql.InvocationHandlerFactory#getTargetClass()
047         */
048        @Override
049        public Class<ResultSet> getTargetClass()
050        {
051                return ResultSet.class;
052        }
053
054        /**
055         * {@inheritDoc}
056         * @see net.sf.hajdbc.sql.InvocationHandlerFactory#createInvocationHandler(java.lang.Object, net.sf.hajdbc.sql.SQLProxy, net.sf.hajdbc.invocation.Invoker, java.util.Map)
057         */
058        @Override
059        public InvocationHandler createInvocationHandler(S statement, SQLProxy<Z, D, S, SQLException> proxy, Invoker<Z, D, S, ResultSet, SQLException> invoker, Map<D, ResultSet> resultSets) throws SQLException
060        {
061                return new ResultSetInvocationHandler<Z, D, S>(statement, proxy, invoker, resultSets, this.transactionContext, this.fileSupport);
062        }
063}
