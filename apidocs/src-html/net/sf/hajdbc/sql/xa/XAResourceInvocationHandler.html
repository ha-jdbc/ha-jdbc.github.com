018package net.sf.hajdbc.sql.xa;
019
020import java.lang.reflect.Method;
021import java.sql.SQLException;
022import java.util.Arrays;
023import java.util.HashSet;
024import java.util.Map;
025import java.util.Set;
026import java.util.SortedMap;
027import java.util.concurrent.ConcurrentHashMap;
028import java.util.concurrent.ConcurrentMap;
029import java.util.concurrent.locks.Lock;
030
031import javax.sql.XAConnection;
032import javax.sql.XADataSource;
033import javax.transaction.xa.XAException;
034import javax.transaction.xa.XAResource;
035import javax.transaction.xa.Xid;
036
037import net.sf.hajdbc.Database;
038import net.sf.hajdbc.DatabaseCluster;
039import net.sf.hajdbc.durability.Durability;
040import net.sf.hajdbc.invocation.InvocationStrategy;
041import net.sf.hajdbc.invocation.InvocationStrategyEnum;
042import net.sf.hajdbc.invocation.Invoker;
043import net.sf.hajdbc.sql.AbstractChildInvocationHandler;
044import net.sf.hajdbc.sql.DurabilityPhaseRegistry;
045import net.sf.hajdbc.sql.SQLProxy;
046import net.sf.hajdbc.util.StaticRegistry;
047import net.sf.hajdbc.util.reflect.Methods;
048
049/**
050 * @author Paul Ferraro
051 *
052 */
053@SuppressWarnings("nls")
054public class XAResourceInvocationHandler extends AbstractChildInvocationHandler<XADataSource, XADataSourceDatabase, XAConnection, SQLException, XAResource, XAException>
055{
056        private static final Set<Method> driverReadMethodSet = Methods.findMethods(XAResource.class, "getTransactionTimeout", "isSameRM");
057        private static final Set<Method> databaseWriteMethodSet = Methods.findMethods(XAResource.class, "setTransactionTimeout");
058        private static final Set<Method> intraTransactionMethodSet = Methods.findMethods(XAResource.class, "prepare", "end", "recover");
059        private static final Method prepareMethod = Methods.getMethod(XAResource.class, "prepare", Xid.class);
060        private static final Method startMethod = Methods.getMethod(XAResource.class, "start", Xid.class, Integer.TYPE);
061        private static final Method commitMethod = Methods.getMethod(XAResource.class, "commit", Xid.class, Boolean.TYPE);
062        private static final Method rollbackMethod = Methods.getMethod(XAResource.class, "rollback", Xid.class);
063        private static final Method forgetMethod = Methods.getMethod(XAResource.class, "forget", Xid.class);
064        private static final Set<Method> endTransactionMethodSet = new HashSet<Method>(Arrays.asList(commitMethod, rollbackMethod, forgetMethod));
065        
066        private static final StaticRegistry<Method, Durability.Phase> phaseRegistry = new DurabilityPhaseRegistry(Arrays.asList(prepareMethod), Arrays.asList(commitMethod), Arrays.asList(rollbackMethod), Arrays.asList(forgetMethod));
067        
068        // Xids are global - so store in static variable
069        private static final ConcurrentMap<Xid, Lock> lockMap = new ConcurrentHashMap<Xid, Lock>();
070        
071        /**
072         * @param connection
073         * @param proxy
074         * @param invoker
075         * @param objectMap
076         * @throws Exception
077         */
078        protected XAResourceInvocationHandler(XAConnection connection, SQLProxy<XADataSource, XADataSourceDatabase, XAConnection, SQLException> proxy, Invoker<XADataSource, XADataSourceDatabase, XAConnection, XAResource, SQLException> invoker, Map<XADataSourceDatabase, XAResource> objectMap)
079        {
080                super(connection, proxy, invoker, XAResource.class, XAException.class, objectMap);
081        }
082
083        /**
084         * @see net.sf.hajdbc.sql.AbstractInvocationHandler#getInvocationStrategy(java.lang.Object, java.lang.reflect.Method, java.lang.Object[])
085         */
086        @Override
087        protected InvocationStrategy getInvocationStrategy(XAResource resource, Method method, Object[] parameters) throws XAException
088        {
089                if (driverReadMethodSet.contains(method))
090                {
091                        return InvocationStrategyEnum.INVOKE_ON_ANY;
092                }
093                
094                if (databaseWriteMethodSet.contains(method))
095                {
096                        return InvocationStrategyEnum.INVOKE_ON_ALL;
097                }
098                
099                boolean start = method.equals(startMethod);
100                boolean end = endTransactionMethodSet.contains(method);
101                
102                if (start || end || method.equals(prepareMethod) || intraTransactionMethodSet.contains(method))
103                {
104                        final InvocationStrategy strategy = end ? InvocationStrategyEnum.END_TRANSACTION_INVOKE_ON_ALL : InvocationStrategyEnum.TRANSACTION_INVOKE_ON_ALL;
105                        
106                        Xid xid = (Xid) parameters[0];
107                        
108                        DatabaseCluster<XADataSource, XADataSourceDatabase> cluster = this.getDatabaseCluster();
109                        
110                        if (start)
111                        {
112                                final Lock lock = cluster.getLockManager().readLock(null);
113                                
114                                // Lock may already exist if we're resuming a suspended transaction
115                                if (lockMap.putIfAbsent(xid, lock) == null)
116                                {
117                                        return new InvocationStrategy()
118                                        {
119                                                @Override
120                                                public <Z, D extends Database<Z>, T, R, E extends Exception> SortedMap<D, R> invoke(SQLProxy<Z, D, T, E> proxy, Invoker<Z, D, T, R, E> invoker) throws E
121                                                {
122                                                        lock.lock();
123                                                        
124                                                        try
125                                                        {
126                                                                return strategy.invoke(proxy, invoker);
127                                                        }
128                                                        catch (Exception e)
129                                                        {
130                                                                lock.unlock();
131
132                                                                throw proxy.getExceptionFactory().createException(e);
133                                                        }
134                                                }
135                                        };
136                                }
137                        }
138                        
139                        Durability.Phase phase = phaseRegistry.get(method);
140                        
141                        if (phase != null)
142                        {
143                                final InvocationStrategy durabilityStrategy = cluster.getDurability().getInvocationStrategy(strategy, phase, xid);
144                                
145                                if (endTransactionMethodSet.contains(method))
146                                {
147                                        final Lock lock = lockMap.remove(xid);
148
149                                        return new InvocationStrategy()
150                                        {
151                                                @Override
152                                                public <Z, D extends Database<Z>, T, R, E extends Exception> SortedMap<D, R> invoke(SQLProxy<Z, D, T, E> proxy, Invoker<Z, D, T, R, E> invoker) throws E
153                                                {
154                                                        try
155                                                        {
156                                                                return durabilityStrategy.invoke(proxy, invoker);
157                                                        }
158                                                        finally
159                                                        {
160                                                                if (lock != null)
161                                                                {
162                                                                        lock.unlock();
163                                                                }
164                                                        }
165                                                }
166                                        };
167                                }
168                                
169                                return durabilityStrategy;
170                        }
171                        
172                        return strategy;
173                }
174                
175                return super.getInvocationStrategy(resource, method, parameters);
176        }
177
178        /**
179         * {@inheritDoc}
180         * @see net.sf.hajdbc.sql.AbstractInvocationHandler#getInvoker(java.lang.Object, java.lang.reflect.Method, java.lang.Object[])
181         */
182        @Override
183        protected <R> Invoker<XADataSource, XADataSourceDatabase, XAResource, R, XAException> getInvoker(XAResource object, Method method, Object[] parameters) throws XAException
184        {
185                Invoker<XADataSource, XADataSourceDatabase, XAResource, R, XAException> invoker = super.getInvoker(object, method, parameters);
186                
187                Durability.Phase phase = phaseRegistry.get(method);
188                
189                if (method.equals(prepareMethod) || endTransactionMethodSet.contains(method))
190                {
191                        return this.getDatabaseCluster().getDurability().getInvoker(invoker, phase, parameters[0], this.getExceptionFactory());
192                }
193                
194                return invoker;
195        }
196
197        /**
198         * @see net.sf.hajdbc.sql.AbstractChildInvocationHandler#close(java.lang.Object, java.lang.Object)
199         */
200        @Override
201        protected void close(XAConnection connection, XAResource resource)
202        {
203                // Do nothing
204        }
205
206        /**
207         * @see net.sf.hajdbc.sql.AbstractInvocationHandler#isRecordable(java.lang.reflect.Method)
208         */
209        @Override
210        protected boolean isRecordable(Method method)
211        {
212                return databaseWriteMethodSet.contains(method);
213        }
