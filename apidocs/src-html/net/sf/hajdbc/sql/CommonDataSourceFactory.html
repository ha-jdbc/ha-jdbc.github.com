018package net.sf.hajdbc.sql;
019
020import java.sql.SQLException;
021import java.util.Hashtable;
022
023import javax.naming.Context;
024import javax.naming.Name;
025import javax.naming.RefAddr;
026import javax.naming.Reference;
027import javax.naming.spi.ObjectFactory;
028
029import net.sf.hajdbc.Database;
030import net.sf.hajdbc.DatabaseCluster;
031import net.sf.hajdbc.DatabaseClusterConfigurationFactory;
032import net.sf.hajdbc.ExceptionType;
033import net.sf.hajdbc.util.Objects;
034import net.sf.hajdbc.util.reflect.ProxyFactory;
035
036/**
037 * @author Paul Ferraro
038 *
039 * @param <Z>
040 * @param <D>
041 */
042public abstract class CommonDataSourceFactory<Z extends javax.sql.CommonDataSource, D extends Database<Z>> implements ObjectFactory, CommonDataSourceInvocationHandlerFactory<Z, D>
043{
044        private final Class<Z> targetClass;
045        
046        /**
047         * @param targetClass
048         */
049        protected CommonDataSourceFactory(Class<Z> targetClass)
050        {
051                this.targetClass = targetClass;
052        }
053
054        /**
055         * @see javax.naming.spi.ObjectFactory#getObjectInstance(java.lang.Object, javax.naming.Name, javax.naming.Context, java.util.Hashtable)
056         */
057        @Override
058        public Object getObjectInstance(Object object, Name name, Context context, Hashtable<?,?> environment) throws Exception
059        {
060                if ((object == null) || !(object instanceof Reference)) return null;
061                
062                Reference reference = (Reference) object;
063                
064                String className = reference.getClassName();
065                
066                if ((className == null) || !className.equals(this.targetClass.getName())) return null;
067                
068                RefAddr idAddr = reference.get(CommonDataSourceReference.CLUSTER);
069                
070                if (idAddr == null) return null;
071                
072                Object idAddrContent = idAddr.getContent();
073                
074                if ((idAddrContent == null) || !(idAddrContent instanceof String)) return null;
075                
076                String id = (String) idAddrContent;
077                
078                RefAddr configAddr = reference.get(CommonDataSourceReference.CONFIG);
079                
080                if (configAddr == null) return null;
081                
082                Object configAddrContent = configAddr.getContent();
083                
084                if ((configAddrContent == null) || !(configAddrContent instanceof byte[])) return null;
085                
086                byte[] config = (byte[]) configAddr.getContent();
087                
088                DatabaseClusterConfigurationFactory<Z, D> factory = Objects.deserialize(config);
089                
090                DatabaseCluster<Z, D> cluster = new DatabaseClusterImpl<Z, D>(id, factory.createConfiguration(), factory);
091                
092                try
093                {
094                        cluster.start();
095                }
096                catch (Exception e)
097                {
098                        cluster.stop();
099
100                        throw ExceptionType.getExceptionFactory(SQLException.class).createException(e);
101                }
102                
103                return ProxyFactory.createProxy(this.targetClass, this.createInvocationHandler(cluster));
104        }
105        
106        /**
107         * {@inheritDoc}
108         * @see net.sf.hajdbc.sql.CommonDataSourceInvocationHandlerFactory#getTargetClass()
109         */
110        @Override
111        public Class<Z> getTargetClass()
112        {
113                return this.targetClass;
114        }
115}
